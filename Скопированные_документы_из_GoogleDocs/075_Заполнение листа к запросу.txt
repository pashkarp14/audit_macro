﻿Option Explicit


'========================================================================================
' Модуль:    modK_Request_Collector
' Версия:    1.2.0  (29.09.2025)
'
' Новое в 1.2.0:
'   - Перед началом работы полностью очищается лист "К запросу".
'   - Расширен список листов-источников (см. SourceSheets()).
'   - При отсутствии листа — подробный лог и продолжение работы.
'
' Новое в 1.1.0:
'   - INCLUDE_LAST3_COLS: False => обрезаем 3 правых столбца при копировании.
'   - Одна толстая внешняя рамка вокруг вставленной таблицы (вместе с шапкой).
'
' Назначение:
'   Собрать таблицы под якорем "К запросу>>>" с ряда исходных листов и
'   вставить их на лист "К запросу" блоками:
'     N. <Заголовок блока>
'     Просим предоставить первичные документы (...) >>>
'     [пустая строка]
'     [таблица]
'
' Производительность:
'   - Копирование одним Range.Copy с полным сохранением форматов ячеек.
'   - Поиск границ: шапка (1 строка под якорем) и "Итого" в столбце якоря.
'
' Требования:
'   - Без MsgBox; LogWrite — Private; использовать функции; без Attribute VB_Name;
'   - Не использовать переменную cDate; после "_" не ставить комментарии.
'========================================================================================


Private Const MODULE_VERSION As String = "1.2.0"
Private Const LOG_SHEET      As String = "Логи"
Private Const DEST_SHEET     As String = "К запросу"
Private Const ANCHOR_TEXT    As String = "К запросу>>>"
Private Const TOTAL_TEXT     As String = "Итого"
Private Const REQ_LINE       As String = _
    "Просим предоставить первичные документы (например, акт, УПД, счет-фактура, товарная накладная, заявка/инвойс, ГТД, ТТН, и иное при наличии) по следующим хозяйственным операциями>>>"


' --------- Настройка: копировать ли последние 3 столбца (True — копируем всё; False — обрезаем справа 3) ---------
Private Const INCLUDE_LAST3_COLS As Boolean = False


'========================== ПУБЛИЧНЫЙ ВХОД ==============================================


Public Sub Заполнить_К_Запросу_Сборка()
    Dim t0 As Double
    Dim prevCalc As XlCalculation
    Dim prevScr As Boolean
    Dim prevEv  As Boolean
    Dim prevDisp As Boolean
    
    t0 = Timer
    On Error GoTo EH
    
    prevCalc = Application.Calculation
    prevScr = Application.ScreenUpdating
    prevEv = Application.EnableEvents
    prevDisp = Application.DisplayAlerts
    
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    Application.DisplayAlerts = False
    Application.Calculation = xlCalculationManual
    
    LogWrite "Старт. Версия=" & MODULE_VERSION & "; INCLUDE_LAST3_COLS=" & CStr(INCLUDE_LAST3_COLS)
    
    ' Приёмник
    Dim wsDest As Worksheet
    Set wsDest = EnsureSheet(DEST_SHEET, False)
    ClearSheetAll wsDest
    LogWrite "Лист '" & DEST_SHEET & "' очищен."
    
    ' Источники
    Dim arrSheets As Variant
    arrSheets = SourceSheets()
    LogWrite "Список исходных листов: " & JoinArray(arrSheets, ", ")
    
    ' Итерация по листам
    Dim i As Long
    Dim sectionN As Long
    Dim nextRow As Long
    sectionN = 0
    nextRow = NextFreeRow(wsDest, 2, False) ' чистый лист — начнём с 1-й строки
    
    For i = LBound(arrSheets) To UBound(arrSheets)
        ProcessOneSheet CStr(arrSheets(i)), wsDest, sectionN, nextRow
    Next i
    
    LogWrite "Готово. Блоков собрано: " & sectionN & "; Время, сек: " & Format(Timer - t0, "0.00")
    
TIDY:
    Application.Calculation = prevCalc
    Application.ScreenUpdating = prevScr
    Application.EnableEvents = prevEv
    Application.DisplayAlerts = prevDisp
    Exit Sub
    
EH:
    LogWrite "ОШИБКА: " & Err.Number & " — " & Err.Description
    Resume TIDY
End Sub


'========================== КОНФИГУРАЦИЯ ===============================================


Private Function SourceSheets() As Variant
    ' Порядок важен: редактируйте по необходимости.
    SourceSheets = Array( _
        "4. Операции без содержания", _
        "7. Праздники", _
        "8. Нетипичные корреспонденции", _
        "9. Ручные со связ. ст", _
        "10. РО по вводу доходов", _
        "11. РО (Random)", _
        "12. Ключевые слова", _
        "13. ПОДФТ" _
    )
End Function


Private Function BlockTitleBySheet(ByVal sheetName As String) As String
    Select Case LCase$(sheetName)
        Case LCase$("4. Операции без содержания")
            BlockTitleBySheet = "Проводки без содержания"
        Case LCase$("7. Праздники")
            BlockTitleBySheet = "Тест проводок в праздничные дни"
        Case LCase$("8. Нетипичные корреспонденции")
            BlockTitleBySheet = "Нетипичные корреспонденции"
        Case LCase$("9. Ручные со связ. ст")
            BlockTitleBySheet = "Ручные проводки со связанными сторонами"
        Case LCase$("10. РО по вводу доходов")
            BlockTitleBySheet = "РО по вводу доходов"
        Case LCase$("11. РО (Random)")
            BlockTitleBySheet = "РО (Random)"
        Case LCase$("12. Ключевые слова")
            BlockTitleBySheet = "Тест по ключевым словам"
        Case LCase$("13. ПОДФТ")
            BlockTitleBySheet = "ПОДФТ"
        Case Else
            BlockTitleBySheet = "Блок: " & sheetName
    End Select
End Function


'========================== ОСНОВНАЯ ЛОГИКА ============================================


Private Sub ProcessOneSheet(ByVal sheetName As String, _
                            ByRef wsDest As Worksheet, _
                            ByRef sectionN As Long, _
                            ByRef nextRow As Long)
    On Error GoTo EH
    
    Dim wsSrc As Worksheet
    Set wsSrc = GetSheetIfExists(sheetName)
    If wsSrc Is Nothing Then
        LogWrite "Пропуск: лист '" & sheetName & "' не найден (это допустимо)."
        Exit Sub
    End If
    
    LogWrite "Обработка листа: '" & sheetName & "'"
    
    ' Якорь
    Dim anchor As Range
    Set anchor = FindFirst(wsSrc, ANCHOR_TEXT)
    If anchor Is Nothing Then
        LogWrite "Пропуск '" & sheetName & "': якорь '" & ANCHOR_TEXT & "' не найден."
        Exit Sub
    End If
    LogWrite "Якорь: адрес=" & anchor.Address(0, 0) & "; кол.=" & anchor.Column
    
    Dim headerRow As Long, dataTop As Long, anchorCol As Long
    headerRow = anchor.Row + 1
    dataTop = headerRow + 1
    anchorCol = anchor.Column
    
    ' Горизонтальные границы по шапке
    Dim firstCol As Long, lastCol As Long
    firstCol = anchorCol
    lastCol = LastUsedColInRow(wsSrc, headerRow)
    If lastCol < firstCol Then
        LogWrite "Пропуск '" & sheetName & "': шапка пустая (row " & headerRow & ")."
        Exit Sub
    End If
    
    Dim origCols As Long
    origCols = lastCol - firstCol + 1
    
    ' Низ: "Итого" вниз по колонке якоря
    Dim cellTotal As Range
    Set cellTotal = FindInColumnDown(wsSrc, anchorCol, TOTAL_TEXT, headerRow)
    If cellTotal Is Nothing Then
        LogWrite "Пропуск '" & sheetName & "': не найдена '" & TOTAL_TEXT & "' в кол. " & anchorCol
        Exit Sub
    End If
    
    Dim dataBottom As Long
    dataBottom = cellTotal.Row - 1
    If dataBottom < headerRow Then
        LogWrite "Пропуск '" & sheetName & "': нет данных между шапкой и 'Итого'."
        Exit Sub
    End If
    
    ' Корректировка правой границы, если обрезаем 3 столбца
    Dim adjLastCol As Long
    adjLastCol = lastCol
    If INCLUDE_LAST3_COLS = False Then
        adjLastCol = lastCol - 3
        If adjLastCol < firstCol Then
            LogWrite "Пропуск '" & sheetName & "': после обрезки нет столбцов (исх=" & origCols & ")."
            Exit Sub
        End If
    End If
    
    ' Диапазон копирования (шапка + данные)
    Dim rngCopy As Range
    Set rngCopy = wsSrc.Range(wsSrc.Cells(headerRow, firstCol), wsSrc.Cells(dataBottom, adjLastCol))
    
    LogWrite "Копирование: " & rngCopy.Address(0, 0) & _
             "; rows=" & rngCopy.Rows.Count & "; cols(исх)=" & origCols & "; cols(встав)=" & rngCopy.Columns.Count
    
    ' ---------- Вставка блока ----------
    sectionN = sectionN + 1
    Dim titleText As String
    titleText = CStr(sectionN) & ". " & BlockTitleBySheet(sheetName)
    
    ' Текстовая шапка блока (колонка B)
    wsDest.Cells(nextRow, 2).Value = titleText
    wsDest.Cells(nextRow + 1, 2).Value = REQ_LINE
    wsDest.Cells(nextRow + 2, 2).Value = ""
    
    ' Таблица
    Dim pasteTopLeft As Range
    Set pasteTopLeft = wsDest.Cells(nextRow + 3, 2)
    rngCopy.Copy pasteTopLeft
    Application.CutCopyMode = False
    
    Dim pasteBR As Range
    Set pasteBR = wsDest.Cells(pasteTopLeft.Row + rngCopy.Rows.Count - 1, _
                               pasteTopLeft.Column + rngCopy.Columns.Count - 1)
    Dim rngPasted As Range
    Set rngPasted = wsDest.Range(pasteTopLeft, pasteBR)
    
    ' Одна толстая внешняя рамка
    DrawThickOuterBorder rngPasted
    
    ' Следующая позиция (2 пустые строки после таблицы)
    nextRow = pasteBR.Row + 2
    
    LogWrite "Вставлено: '" & titleText & "'; область " & rngPasted.Address(0, 0)
    Exit Sub
    
EH:
    LogWrite "ОШИБКА лист '" & sheetName & "': " & Err.Number & " — " & Err.Description
End Sub


'========================== ПОИСК/ОБЛАСТИ ==============================================


Private Function FindFirst(ByRef ws As Worksheet, ByVal whatText As String) As Range
    Dim rng As Range
    On Error Resume Next
    Set rng = ws.Cells.Find(What:=whatText, LookIn:=xlValues, LookAt:=xlPart, _
                            SearchOrder:=xlByRows, SearchDirection:=xlNext, MatchCase:=False)
    On Error GoTo 0
    If Not rng Is Nothing Then
        Set FindFirst = rng
    Else
        Set FindFirst = Nothing
    End If
End Function


Private Function FindInColumnDown(ByRef ws As Worksheet, ByVal colIndex As Long, _
                                  ByVal whatText As String, ByVal startRow As Long) As Range
    Dim rng As Range
    Dim lastR As Long
    lastR = ws.Cells(ws.Rows.Count, colIndex).End(xlUp).Row
    If lastR < startRow + 1 Then
        Set FindInColumnDown = Nothing
        Exit Function
    End If
    On Error Resume Next
    Set rng = ws.Range(ws.Cells(startRow + 1, colIndex), ws.Cells(lastR, colIndex)) _
                 .Find(What:=whatText, LookIn:=xlValues, LookAt:=xlPart, _
                       SearchOrder:=xlByRows, SearchDirection:=xlNext, MatchCase:=False)
    On Error GoTo 0
    If Not rng Is Nothing Then
        Set FindInColumnDown = rng
    Else
        Set FindInColumnDown = Nothing
    End If
End Function


Private Function LastUsedColInRow(ByRef ws As Worksheet, ByVal rowIndex As Long) As Long
    LastUsedColInRow = ws.Cells(rowIndex, ws.Columns.Count).End(xlToLeft).Column
End Function


'========================== ЛИСТ/СТРОКИ/ОЧИСТКА ========================================


Private Function EnsureSheet(ByVal sheetName As String, ByVal clearForUse As Boolean) As Worksheet
    Dim ws As Worksheet
    Set ws = GetSheetIfExists(sheetName)
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        ws.Name = sheetName
        LogWrite "Создан лист '" & sheetName & "'."
    End If
    If clearForUse Then
        ClearSheetAll ws
    End If
    Set EnsureSheet = ws
End Function


Private Sub ClearSheetAll(ByRef ws As Worksheet)
    On Error Resume Next
    ws.Cells.Clear
    ws.ShowAllData
    On Error GoTo 0
    On Error Resume Next
    ws.AutoFilterMode = False
    On Error GoTo 0
    On Error Resume Next
    Dim shp As Shape
    For Each shp In ws.Shapes
        shp.Delete
    Next shp
    On Error GoTo 0
End Sub


Private Function GetSheetIfExists(ByVal sheetName As String) As Worksheet
    On Error Resume Next
    Set GetSheetIfExists = ThisWorkbook.Worksheets(sheetName)
    On Error GoTo 0
End Function


Private Function NextFreeRow(ByRef ws As Worksheet, ByVal byCol As Long, ByVal addSpacer As Boolean) As Long
    Dim r As Long
    r = ws.Cells(ws.Rows.Count, byCol).End(xlUp).Row
    If r < 1 Then r = 1
    If LenB(Trim$(CStr(ws.Cells(1, byCol).Value))) = 0 And ws.UsedRange.Cells.Count = 1 Then
        NextFreeRow = 1
    Else
        If addSpacer Then
            NextFreeRow = r + 2
        Else
            NextFreeRow = r + 1
        End If
    End If
End Function


Private Function JoinArray(ByVal arr As Variant, ByVal sep As String) As String
    Dim i As Long
    Dim tmp As String
    If IsArray(arr) Then
        For i = LBound(arr) To UBound(arr)
            If i > LBound(arr) Then tmp = tmp & sep
            tmp = tmp & CStr(arr(i))
        Next i
    Else
        tmp = CStr(arr)
    End If
    JoinArray = tmp
End Function


'========================== РАМКИ ======================================================


Private Sub DrawThickOuterBorder(ByRef rng As Range)
    On Error Resume Next
    With rng.Borders(xlEdgeLeft)
        .LineStyle = xlContinuous
        .Weight = xlThick
    End With
    With rng.Borders(xlEdgeRight)
        .LineStyle = xlContinuous
        .Weight = xlThick
    End With
    With rng.Borders(xlEdgeTop)
        .LineStyle = xlContinuous
        .Weight = xlThick
    End With
    With rng.Borders(xlEdgeBottom)
        .LineStyle = xlContinuous
        .Weight = xlThick
    End With
    On Error GoTo 0
End Sub


'========================== ЛОГИРОВАНИЕ ================================================


Private Sub LogWrite(ByVal msg As String)
    Dim ws As Worksheet
    Dim r As Long
    Dim ts As String
    
    On Error GoTo EH
    Set ws = GetSheetIfExists(LOG_SHEET)
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(Before:=ThisWorkbook.Worksheets(1))
        ws.Name = LOG_SHEET
    End If
    
    r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If r < 1 Then r = 1
    
    ts = Format(Now, "dd.mm.yyyy HH:nn:ss")
    If LenB(Trim$(CStr(ws.Cells(1, 1).Value))) = 0 And ws.UsedRange.Cells.Count = 1 Then
        r = 1
    Else
        r = r + 1
    End If
    
    ws.Cells(r, 1).Value = ts
    ws.Cells(r, 2).Value = "K_Request_Collector"
    ws.Cells(r, 3).Value = MODULE_VERSION
    ws.Cells(r, 4).Value = msg
    Exit Sub
EH:
    ' Тихая деградация логирования
End Sub
