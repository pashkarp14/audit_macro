﻿Option Explicit


'========================================================================================
' Модуль:    modOps_Keywords_Population
' Версия:    1.5.0  (23.09.2025)
'
' Что нового vs 1.4.0:
'   - В области вставки исключены столбцы: "Связанные?", "Ответственный", "Протестировано?".
'     Порядок прочих колонок — как на образце; "Проводка" становится последним столбцом.
'   - Формула "Итого" подстраивается к позиции "Сумма" среди оставленных колонок.
'   - Доп. логирование по выбранным/исключённым столбцам.
'
' Условия отбора:
'   • "Протестировано?" — пусто ИЛИ (если INCLUDE_OWN_MARKED=True) содержит OWN_TEST_LABEL;
'   • в колонках из KW_SearchColumnsSyns() найдено хотя бы одно ключевое слово
'     со страницы "Слова для теста" (колонка A).
'
' Оптимизация:
'   • Индексы шапок — на каждом листе популяции.
'   • Фильтрация и сбор данных — массивами.
'   • Форматы — только в области вставки; целые столбцы не трогаем.
'========================================================================================


Private Const MacroVersion       As String = "1.5.0"
Private Const LOG_SHEET          As String = "Логи"


' ---------- ЦЕЛЕВАЯ ОБЛАСТЬ ВЫВОДА ----------
Private Const DST_SHEET          As String = "9. Ключевые слова"
Private Const ANCHOR_COL         As Long = 2
Private Const ANCHOR_TEXT        As String = "Проводки с ключевыми словами>>>"
Private Const TOTAL_TEXT         As String = "Итого"
Private Const NUM_FMT            As String = "# ##0;-# ##0;-"


' ---------- Источники ----------
Private Const POP_BASE           As String = "Популяция для тестирования"
Private Const POP_PART_SUFF      As String = "_Ч"


' ---------- Список ключевых слов ----------
Private Const KW_SHEET           As String = "Слова для теста"
Private Const KW_COL             As Long = 1


' ---------- Метка теста / «свои помеченные» ----------
Private Const OWN_TEST_LABEL     As String = "См. ключевые слова"
Private Const INCLUDE_OWN_MARKED As Boolean = True


' ---------- Синонимы ----------
Private Function SynTested() As Variant
    SynTested = Array("Протестировано?", " протестировано?", "протестировано? ", "Протестировано", " протестировано", "протестировано ")
End Function


' Поисковые колонки (настройка — можно расширять)
Private Function KW_SearchColumnsSyns() As Variant
    KW_SearchColumnsSyns = Array( _
        Array("Содержание", "Описание", "Комментарий", "Примечание", "Назначение", "Текст") _
    )
End Function


'========================================================================================
' ТОЧКА ВХОДА
'========================================================================================
Public Sub Заполнить_РО_ключевые_слова_популяция()
    Dim t0 As Double: t0 = Timer
    Dim appCalc As XlCalculation
    Dim scrUp As Boolean, evt As Boolean, alr As Boolean
    Dim errStep As String
    Dim tStep As Double


    appCalc = Application.Calculation
    scrUp = Application.ScreenUpdating
    evt = Application.EnableEvents
    alr = Application.DisplayAlerts


    On Error GoTo EH
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    Application.DisplayAlerts = False
    Application.Calculation = xlCalculationManual


    LogWrite KW_NowStr() & vbTab & "Старт" & vbTab & _
            "Заполнить_РО_ключевые_слова_популяция v" & MacroVersion


    Dim wb As Workbook: Set wb = ThisWorkbook


    ' ---------- 0) Целевой лист и якорь ----------
    errStep = "Поиск листа назначения и якоря"
    KW_StepStart errStep, tStep


    Dim wsDst As Worksheet
    Set wsDst = KW_EnsureSheet(wb, DST_SHEET)


    Dim createdBlock As Boolean: createdBlock = False
    Dim anchorRow As Long, headerTop As Long, dataTop As Long, totalRow As Long, dataBottom As Long


    KW_EnsureAnchorAndTotal wsDst, ANCHOR_COL, ANCHOR_TEXT, TOTAL_TEXT, createdBlock


    anchorRow = KW_FindFirstInColumnExact(wsDst, ANCHOR_COL, ANCHOR_TEXT)
    If anchorRow = 0 Then Err.Raise 5, , "Якорь не найден и не удалось создать."


    headerTop = anchorRow + 1
    dataTop = headerTop + 1


    totalRow = KW_FindFirstBelowInColumnExact(wsDst, ANCHOR_COL, TOTAL_TEXT, dataTop)
    If totalRow = 0 Or totalRow <= dataTop Then Err.Raise 5, , "Метка 'Итого' не найдена ниже области данных."


    dataBottom = totalRow - 1


    LogWrite KW_NowStr() & vbTab & "? Приёмник" & vbTab & _
        "AnchorRow=" & anchorRow & "; HeaderTop=" & headerTop & "; DataTop=" & dataTop & _
        "; TotalRow=" & totalRow & "; BlockCreated=" & IIf(createdBlock, "Yes", "No")
    KW_StepEnd errStep, tStep


    ' ---------- 1) Ключевые слова ----------
    errStep = "Чтение ключевых слов"
    KW_StepStart errStep, tStep
    Dim kwList As Variant, kwCount As Long
    kwList = KW_ReadKeywords(wb, kwCount)
    LogWrite KW_NowStr() & vbTab & "Ключевых слов (уник.)" & vbTab & kwCount
    If kwCount = 0 Then
        LogWrite KW_NowStr() & vbTab & "Внимание" & vbTab & "Список ключевых слов пуст — выход"
        GoTo FinishOK
    End If
    KW_StepEnd errStep, tStep


    ' ---------- 2) Листы популяции ----------
    errStep = "Поиск листов популяции (base ИЛИ parts)"
    KW_StepStart errStep, tStep
    Dim popSheets As Collection
    Set popSheets = KW_GetPopulationSheets(wb)
    If popSheets Is Nothing Or popSheets.Count = 0 Then Err.Raise 5, , "Листы популяции не найдены"
    LogWrite KW_NowStr() & vbTab & "Найдено листов популяции" & vbTab & popSheets.Count
    KW_StepEnd errStep, tStep


    ' ---------- 3) Образец шапки + набор выводимых колонок (исключая 3 последних) ----------
    errStep = "Определение шапки образца и выбор выводимых колонок"
    KW_StepStart errStep, tStep
    Dim wsSample As Worksheet: Set wsSample = popSheets(1)
    Dim lastColSample As Long
    lastColSample = wsSample.Cells(1, wsSample.Columns.Count).End(xlToLeft).Column


    ' Найдём индексы "Связанные?"/"Ответственный"/"Протестировано?"
    Dim hdrLinked As String, hdrResp As String, hdrTested As String, hdrSum As String
    Dim cLinked As Long, cResp As Long, cTestedS As Long, cSumSample As Long


    cLinked = KW_FindHeaderSyn(wsSample, Array("Связанные?", " связанные?", "связанные? "), 1, hdrLinked)
    cResp = KW_FindHeaderSyn(wsSample, Array("Ответственный", " ответственный", "ответственный "), 1, hdrResp)
    cTestedS = KW_FindHeaderSyn(wsSample, SynTested(), 1, hdrTested)
    cSumSample = KW_FindHeaderSyn(wsSample, Array("Сумма", " сумма", "сумма ", "Сумма абс", "Сумма абс.", "Сумма абсолютная", "Сумма (абс)"), 1, hdrSum)


    ' Список колонок к выводу — как на образце, но исключаем найденные 3
    Dim keepCols As New Collection
    Dim j As Long
    For j = 1 To lastColSample
        If j <> cLinked And j <> cResp And j <> cTestedS Then
            keepCols.Add j
        End If
    Next j


    Dim outCols As Long: outCols = keepCols.Count


    ' Имена шапок выводимых колонок
    Dim keepHdrNames() As String
    ReDim keepHdrNames(1 To outCols)
    For j = 1 To outCols
        keepHdrNames(j) = CStr(wsSample.Cells(1, CLng(keepCols(j))).Value)
    Next j


    ' Позиция "Сумма" среди выводимых (для "Итого")
    Dim posSumOut As Long: posSumOut = KW_PosOfColInKeep(keepCols, cSumSample)


    LogWrite KW_NowStr() & vbTab & "? Шапка/колонки" & vbTab & _
        "Всего=" & lastColSample & "; К_выводу=" & outCols & _
        "; Исключены: Linked=" & KW_PosDesc(cLinked) & _
        "; Resp=" & KW_PosDesc(cResp) & "; Tested=" & KW_PosDesc(cTestedS) & _
        "; Sum=c" & KW_PosDesc(cSumSample) & " -> OutPos=" & posSumOut
    KW_StepEnd errStep, tStep


    ' ---------- 4) Подсчёт строк к выводу ----------
    errStep = "Подсчёт строк, проходящих условия"
    KW_StepStart errStep, tStep
    Dim totalRows As Long: totalRows = 0


    Dim shIdx As Long
    For shIdx = 1 To popSheets.Count
        Dim ws As Worksheet: Set ws = popSheets(shIdx)


        Dim cTested As Long, hdrT As String
        cTested = KW_FindHeaderSyn(ws, SynTested(), 1, hdrT)
        If cTested = 0 Then
            LogWrite KW_NowStr() & vbTab & "Пропуск листа (нет 'Протестировано?'): " & ws.Name
            GoTo NextSheetCount
        End If


        Dim searchSyns As Variant: searchSyns = KW_SearchColumnsSyns()
        Dim searchCols As New Collection
        Dim sHdr As String, c As Long, b As Long
        For b = LBound(searchSyns) To UBound(searchSyns)
            c = KW_FindHeaderSyn(ws, searchSyns(b), 1, sHdr)
            If c > 0 Then searchCols.Add c
        Next b
        If searchCols.Count = 0 Then
            LogWrite KW_NowStr() & vbTab & "Пропуск листа (нет колонок для поиска): " & ws.Name
            GoTo NextSheetCount
        End If


        Dim dynCols() As Long
        ReDim dynCols(1 To searchCols.Count + 1)
        dynCols(1) = cTested
        For b = 1 To searchCols.Count
            dynCols(1 + b) = CLng(searchCols(b))
        Next b


        Dim lr As Long
        lr = KW_LastDataRowOnCols(ws, dynCols)
        If lr < 2 Then GoTo NextSheetCount


        Dim n As Long: n = lr - 1
        Dim arrT As Variant: arrT = KW_Read2D(ws, 2, cTested, lr, cTested)


        Dim arrSearch() As Variant: ReDim arrSearch(1 To searchCols.Count)
        For b = 1 To searchCols.Count
            arrSearch(b) = KW_Read2D(ws, 2, CLng(searchCols(b)), lr, CLng(searchCols(b)))
        Next b


        Dim r As Long, hasKW As Boolean, kw As Variant
        For r = 1 To n
            If KW_AcceptTestedCell(arrT(r, 1), INCLUDE_OWN_MARKED, OWN_TEST_LABEL) Then
                hasKW = False
                Dim bIdx As Long
                For bIdx = 1 To searchCols.Count
                    Dim cellText As String: cellText = KW_NormTextForSearch(arrSearch(bIdx)(r, 1))
                    If LenB(cellText) > 0 Then
                        For Each kw In kwList
                            If InStr(1, cellText, CStr(kw), vbTextCompare) > 0 Then
                                hasKW = True
                                Exit For
                            End If
                        Next kw
                        If hasKW Then Exit For
                    End If
                Next bIdx
                If hasKW Then totalRows = totalRows + 1
            End If
        Next r


        LogWrite KW_NowStr() & vbTab & "Подсчёт на листе" & vbTab & ws.Name & ": к добавлению (накоп.)=" & totalRows
NextSheetCount:
    Next shIdx
    LogWrite KW_NowStr() & vbTab & "? Подсчёт" & vbTab & "Всего строк к копированию=" & totalRows
    KW_StepEnd errStep, tStep


    ' ---------- 5) Очистка области и подгонка строк ----------
    errStep = "Очистка области/подгонка количества строк"
    KW_StepStart errStep, tStep
    KW_ClearBorders wsDst, wsDst.Range(wsDst.Cells(headerTop, ANCHOR_COL), _
                                       wsDst.Cells(IIf(dataBottom >= headerTop, dataBottom, headerTop), ANCHOR_COL + outCols - 1))


    If dataBottom >= dataTop Then
        wsDst.Range(wsDst.Cells(dataTop, ANCHOR_COL), wsDst.Cells(dataBottom, ANCHOR_COL + outCols - 1)).ClearContents
    End If
    wsDst.Range(wsDst.Cells(headerTop, ANCHOR_COL), wsDst.Cells(headerTop, ANCHOR_COL + outCols - 1)).ClearContents


    Dim need As Long: need = IIf(totalRows > 0, totalRows, 0)
    Dim nRowsDst As Long: nRowsDst = dataBottom - dataTop + 1


    If nRowsDst < need Then
        Dim insN As Long: insN = need - nRowsDst
        If insN > 0 Then
            wsDst.Rows(dataBottom).Resize(insN).Insert xlShiftDown
            totalRow = totalRow + insN
            dataBottom = dataTop + need - 1
            LogWrite KW_NowStr() & vbTab & "Вставлено строк" & vbTab & insN
        End If
    ElseIf nRowsDst > need Then
        Dim delN As Long: delN = nRowsDst - need
        If delN > 0 Then
            wsDst.Rows(dataTop + need).Resize(delN).Delete xlShiftUp
            totalRow = totalRow - delN
            dataBottom = dataTop + need - 1
            LogWrite KW_NowStr() & vbTab & "Удалено лишних строк" & vbTab & delN
        End If
    Else
        LogWrite KW_NowStr() & vbTab & "Подгонка строк" & vbTab & "Не требуется"
    End If
    KW_StepEnd errStep, tStep


    ' ---------- 6) Шапка (значения + форматы) ----------
    errStep = "Вывод шапки по образцу (без 3 исключённых)"
    KW_StepStart errStep, tStep
    For j = 1 To outCols
        Dim srcCol As Long: srcCol = CLng(keepCols(j))
        Dim dstCol As Long: dstCol = ANCHOR_COL + j - 1


        wsDst.Cells(headerTop, dstCol).Value = wsSample.Cells(1, srcCol).Value


        wsSample.Cells(1, srcCol).Copy
        wsDst.Cells(headerTop, dstCol).PasteSpecial xlPasteFormats
    Next j
    Application.CutCopyMode = False
    KW_ThickBox wsDst.Range(wsDst.Cells(headerTop, ANCHOR_COL), wsDst.Cells(headerTop, ANCHOR_COL + outCols - 1))
    KW_StepEnd errStep, tStep


    ' ---------- 7) Сбор данных и вывод массивом (мэппинг по именам) ----------
    errStep = "Сбор данных в массив и вывод"
    KW_StepStart errStep, tStep
    Dim outArr() As Variant
    If totalRows > 0 Then
        ReDim outArr(1 To totalRows, 1 To outCols)
    Else
        ReDim outArr(1 To 1, 1 To outCols)
    End If


    Dim writeRow As Long: writeRow = 1


    For shIdx = 1 To popSheets.Count
        Dim wsSrc As Worksheet: Set wsSrc = popSheets(shIdx)


        Dim cTested2 As Long, hdrTested2 As String
        cTested2 = KW_FindHeaderSyn(wsSrc, SynTested(), 1, hdrTested2)
        If cTested2 = 0 Then GoTo NextSheetFill


        Dim searchSyns2 As Variant: searchSyns2 = KW_SearchColumnsSyns()
        Dim searchCols2 As New Collection
        Dim c2 As Long, hdr2 As String, b2 As Long
        For b2 = LBound(searchSyns2) To UBound(searchSyns2)
            c2 = KW_FindHeaderSyn(wsSrc, searchSyns2(b2), 1, hdr2)
            If c2 > 0 Then searchCols2.Add c2
        Next b2
        If searchCols2.Count = 0 Then GoTo NextSheetFill


        Dim lr2 As Long
        lr2 = KW_LastDataRowOnCols(wsSrc, Array(cTested2, CLng(searchCols2(1))))
        If lr2 < 2 Then GoTo NextSheetFill


        Dim hdrMap As Object: Set hdrMap = KW_BuildHeaderIndex(wsSrc, 1)


        Dim n2 As Long: n2 = lr2 - 1
        Dim arrT2 As Variant: arrT2 = KW_Read2D(wsSrc, 2, cTested2, lr2, cTested2)


        Dim arrS2() As Variant: ReDim arrS2(1 To searchCols2.Count)
        For b2 = 1 To searchCols2.Count
            arrS2(b2) = KW_Read2D(wsSrc, 2, CLng(searchCols2(b2)), lr2, CLng(searchCols2(b2)))
        Next b2


        Dim maxC As Long: maxC = wsSrc.Cells(1, wsSrc.Columns.Count).End(xlToLeft).Column
        Dim allBlock As Variant
        allBlock = wsSrc.Range(wsSrc.Cells(1, 1), wsSrc.Cells(lr2, maxC)).Value2


        Dim r2 As Long, hasKW2 As Boolean, kw2 As Variant
        For r2 = 1 To n2
            If KW_AcceptTestedCell(arrT2(r2, 1), INCLUDE_OWN_MARKED, OWN_TEST_LABEL) Then
                hasKW2 = False
                Dim bb As Long
                For bb = 1 To searchCols2.Count
                    Dim cellText2 As String: cellText2 = KW_NormTextForSearch(arrS2(bb)(r2, 1))
                    If LenB(cellText2) > 0 Then
                        For Each kw2 In kwList
                            If InStr(1, cellText2, CStr(kw2), vbTextCompare) > 0 Then
                                hasKW2 = True
                                Exit For
                            End If
                        Next kw2
                        If hasKW2 Then Exit For
                    End If
                Next bb


                If hasKW2 Then
                    Dim j3 As Long
                    For j3 = 1 To outCols
                        Dim wantHdr As String, wantKey As String, srcC As Long, rel As Long
                        wantHdr = keepHdrNames(j3)
                        wantKey = KW_NormalizeHeaderText(wantHdr)
                        If hdrMap.Exists(wantKey) Then
                            srcC = CLng(hdrMap(wantKey))
                            rel = srcC ' allBlock начинается с колонки 1
                            outArr(writeRow, j3) = allBlock(r2 + 1, rel)
                        Else
                            outArr(writeRow, j3) = Empty
                        End If
                    Next j3
                    writeRow = writeRow + 1
                End If
            End If


            If (r2 Mod 200000) = 0 Then DoEvents
        Next r2


        LogWrite KW_NowStr() & vbTab & "Выборка на листе" & vbTab & wsSrc.Name & ": накоплено=" & (writeRow - 1)
NextSheetFill:
    Next shIdx


    If totalRows > 0 Then
        wsDst.Range(wsDst.Cells(dataTop, ANCHOR_COL), wsDst.Cells(dataTop + totalRows - 1, ANCHOR_COL + outCols - 1)).Value = outArr
        KW_ThickBox wsDst.Range(wsDst.Cells(dataTop, ANCHOR_COL), wsDst.Cells(dataTop + totalRows - 1, ANCHOR_COL + outCols - 1))
    End If
    KW_StepEnd errStep, tStep


    ' ---------- 8) Форматы данных по образцу ----------
    errStep = "Форматирование данных (NumberFormatLocal по образцу)"
    KW_StepStart errStep, tStep
    If totalRows > 0 Then
        Dim dstCol2 As Long, srcCol2 As Long
        Dim rngCol As Range
        For j = 1 To outCols
            dstCol2 = ANCHOR_COL + j - 1
            srcCol2 = CLng(keepCols(j))
            Set rngCol = wsDst.Range(wsDst.Cells(dataTop, dstCol2), wsDst.Cells(dataTop + totalRows - 1, dstCol2))
            On Error Resume Next
            rngCol.NumberFormatLocal = wsSample.Cells(2, srcCol2).NumberFormatLocal
            On Error GoTo 0
        Next j
    End If
    KW_StepEnd errStep, tStep


    ' ---------- 9) Формула «Итого» под «Сумма» ----------
    errStep = "Формула суммы под столбцом 'Сумма'"
    KW_StepStart errStep, tStep
    If posSumOut > 0 Then
        Dim dstSumCol As Long: dstSumCol = ANCHOR_COL + posSumOut - 1
        If totalRows > 0 Then
            Dim rngSum As Range
            Set rngSum = wsDst.Range(wsDst.Cells(dataTop, dstSumCol), wsDst.Cells(dataTop + totalRows - 1, dstSumCol))
            wsDst.Cells(totalRow, dstSumCol).FormulaLocal = "=СУММ(" & rngSum.Address(False, False) & ")"
        Else
            wsDst.Cells(totalRow, dstSumCol).FormulaLocal = "=0"
        End If
        wsDst.Cells(totalRow, dstSumCol).NumberFormatLocal = NUM_FMT
    Else
        LogWrite KW_NowStr() & vbTab & "Предупреждение" & vbTab & "В наборе выводимых колонок нет 'Сумма' — формула не выведена"
    End If
    KW_StepEnd errStep, tStep


FinishOK:
    LogWrite KW_NowStr() & vbTab & "Готово" & vbTab & _
            "Строк всего=" & totalRows & "; Колонок (вывод)=" & outCols & _
            "; Время, сек=" & Format$(Timer - t0, "0.00")


Tidy:
    Application.ScreenUpdating = scrUp
    Application.EnableEvents = evt
    Application.DisplayAlerts = alr
    Application.Calculation = appCalc
    Exit Sub


EH:
    LogWrite KW_NowStr() & vbTab & "Ошибка" & vbTab & "Step='" & errStep & "'; " & Err.Number & " — " & Err.Description
    Resume Tidy
End Sub


'========================================================================================

Private Sub KW_StepStart(ByVal stepName As String, ByRef tStart As Double)
    tStart = Timer
    LogWrite KW_NowStr() & vbTab & ">> " & stepName
End Sub


Private Sub KW_StepEnd(ByVal stepName As String, ByRef tStart As Double)
    LogWrite KW_NowStr() & vbTab & "<< " & stepName & "  (" & Format$(Timer - tStart, "0.00") & " сек)"
End Sub


Private Function KW_PosOfColInKeep(ByVal keepCols As Collection, ByVal colIdx As Long) As Long
    Dim i As Long
    For i = 1 To keepCols.Count
        If CLng(keepCols(i)) = colIdx Then KW_PosOfColInKeep = i: Exit Function
    Next i
End Function


Private Function KW_PosDesc(ByVal c As Long) As String
    If c > 0 Then KW_PosDesc = "c" & CStr(c) Else KW_PosDesc = "нет"
End Function


' --- ключевые слова ---
Private Function KW_ReadKeywords(ByVal wb As Workbook, ByRef cnt As Long) As Variant
    Dim ws As Worksheet
    On Error Resume Next: Set ws = wb.Worksheets(KW_SHEET): On Error GoTo 0
    If ws Is Nothing Then cnt = 0: Exit Function


    Dim lastRow As Long: lastRow = ws.Cells(ws.Rows.Count, KW_COL).End(xlUp).Row
    If lastRow < 1 Then cnt = 0: Exit Function


    Dim d As Object: Set d = CreateObject("Scripting.Dictionary"): d.CompareMode = vbTextCompare
    Dim arr As Variant: arr = KW_Read2D(ws, 1, KW_COL, lastRow, KW_COL)


    Dim r As Long, s As String
    For r = 1 To lastRow
        s = KW_NormKeyword(CStr(arr(r, 1)))
        If LenB(s) > 0 Then If Not d.Exists(s) Then d.Add s, True
    Next r


    cnt = d.Count
    If cnt > 0 Then KW_ReadKeywords = d.Keys
End Function


' --- чтение 2D даже для 1x1 ---
Private Function KW_Read2D(ByVal ws As Worksheet, ByVal r1 As Long, ByVal c1 As Long, ByVal r2 As Long, ByVal c2 As Long) As Variant
    Dim v As Variant
    v = ws.Range(ws.Cells(r1, c1), ws.Cells(r2, c2)).Value2
    If (r2 - r1 + 1) = 1 And (c2 - c1 + 1) = 1 Then
        Dim a As Variant: ReDim a(1 To 1, 1 To 1)
        a(1, 1) = v
        KW_Read2D = a
    Else
        KW_Read2D = v
    End If
End Function


' --- источники: base ИЛИ parts ---
Private Function KW_GetPopulationSheets(ByVal wb As Workbook) As Collection
    Dim col As New Collection, sh As Worksheet, foundBase As Boolean
    For Each sh In wb.Worksheets
        If sh.Name = POP_BASE Then col.Add sh: foundBase = True: Exit For
    Next sh
    If foundBase Then Set KW_GetPopulationSheets = col: Exit Function


    For Each sh In wb.Worksheets
        If Left$(sh.Name, Len(POP_BASE & POP_PART_SUFF)) = POP_BASE & POP_PART_SUFF Then col.Add sh
    Next sh
    Set KW_GetPopulationSheets = col
End Function


' --- поиск шапки по синонимам ---
Private Function KW_FindHeaderSyn(ByVal ws As Worksheet, ByVal syns As Variant, ByVal hdrRow As Long, _
                                  ByRef matchedHeader As String) As Long
    Dim lastCol As Long, c As Long, i As Long, head As String
    matchedHeader = ""
    lastCol = ws.Cells(hdrRow, ws.Columns.Count).End(xlToLeft).Column
    For c = 1 To lastCol
        head = KW_NormalizeHeaderText(CStr(ws.Cells(hdrRow, c).Value))
        If head <> "" Then
            For i = LBound(syns) To UBound(syns)
                If head = KW_NormalizeHeaderText(CStr(syns(i))) Then
                    matchedHeader = CStr(ws.Cells(hdrRow, c).Value)
                    KW_FindHeaderSyn = c
                    Exit Function
                End If
            Next i
        End If
    Next c
End Function


Private Function KW_LastDataRowOnCols(ByVal ws As Worksheet, ByVal cols As Variant) As Long
    Dim lr As Long: lr = 0
    Dim j As Long, lrj As Long, c As Long
    For j = LBound(cols) To UBound(cols)
        c = CLng(cols(j))
        If c > 0 Then
            lrj = ws.Cells(ws.Rows.Count, c).End(xlUp).Row
            If lrj > lr Then lr = lrj
        End If
    Next j
    If lr < 2 Then lr = 2
    KW_LastDataRowOnCols = lr
End Function


' --- карта шапки: нормализованное имя -> колонка ---
Private Function KW_BuildHeaderIndex(ByVal ws As Worksheet, ByVal hdrRow As Long) As Object
    Dim dict As Object: Set dict = CreateObject("Scripting.Dictionary")
    dict.CompareMode = vbTextCompare
    Dim lastCol As Long: lastCol = ws.Cells(hdrRow, ws.Columns.Count).End(xlToLeft).Column
    Dim c As Long, key As String, raw As String
    For c = 1 To lastCol
        raw = CStr(ws.Cells(hdrRow, c).Value)
        key = KW_NormalizeHeaderText(raw)
        If Len(key) > 0 Then If Not dict.Exists(key) Then dict.Add key, c
    Next c
    Set KW_BuildHeaderIndex = dict
End Function


'============================== ТЕКСТ/ПОИСК/ФОРМАТ =============================


Private Function KW_NormalizeHeaderText(ByVal s As String) As String
    Dim nb As String: nb = Chr$(160)
    s = Replace$(s, nb, " ")
    s = Replace$(s, vbTab, " ")
    s = Replace$(s, vbCr, " ")
    s = Replace$(s, vbLf, " ")
    s = Replace$(s, "ё", "е")
    s = Replace$(s, "Ё", "Е")
    s = Trim$(s)
    Do While InStr(s, "  ") > 0
        s = Replace$(s, "  ", " ")
    Loop
    KW_NormalizeHeaderText = LCase$(s)
End Function


Private Function KW_AcceptTestedCell(ByVal v As Variant, ByVal includeOwn As Boolean, ByVal ownLabel As String) As Boolean
    If KW_IsEmptyOrTrimEmpty(v) Then KW_AcceptTestedCell = True: Exit Function
    If includeOwn Then
        Dim s As String
        On Error Resume Next: s = CStr(v): On Error GoTo 0
        s = Trim$(s)
        If LenB(s) > 0 Then
            If InStr(1, s, ownLabel, vbTextCompare) > 0 Then KW_AcceptTestedCell = True: Exit Function
        End If
    End If
End Function


Private Function KW_IsEmptyOrTrimEmpty(ByVal v As Variant) As Boolean
    If IsEmpty(v) Then KW_IsEmptyOrTrimEmpty = True: Exit Function
    Dim s As String, nb As String: nb = Chr$(160)
    On Error Resume Next: s = CStr(v): On Error GoTo 0
    s = Replace$(s, nb, " ")
    s = Replace$(s, vbTab, " ")
    s = Replace$(s, vbCr, " ")
    s = Replace$(s, vbLf, " ")
    s = Trim$(s)
    Do While InStr(s, "  ") > 0
        s = Replace$(s, "  ", " ")
    Loop
    KW_IsEmptyOrTrimEmpty = (LenB(s) = 0)
End Function

'============================== ОБЛАСТЬ ВЫВОДА (якорь) =========================


Private Function KW_EnsureSheet(ByVal wb As Workbook, ByVal sheetName As String) As Worksheet
    On Error Resume Next
    Dim ws As Worksheet: Set ws = wb.Worksheets(sheetName)
    On Error GoTo 0
    If ws Is Nothing Then
        Set ws = wb.Worksheets.Add(After:=wb.Worksheets(wb.Worksheets.Count))
        ws.Name = sheetName
        LogWrite KW_NowStr() & vbTab & "Создан лист" & vbTab & sheetName
    End If
    Set KW_EnsureSheet = ws
End Function


Private Sub KW_EnsureAnchorAndTotal(ByVal ws As Worksheet, ByVal colIdx As Long, _
                                    ByVal anchorText As String, ByVal totalText As String, _
                                    ByRef created As Boolean)
    created = False
    Dim aRow As Long
    aRow = KW_FindFirstInColumnExact(ws, colIdx, anchorText)
    If aRow = 0 Then
        ws.Cells(2, colIdx).Value = anchorText
        created = True
        LogWrite KW_NowStr() & vbTab & "Поставлен якорь" & vbTab & "R2C" & colIdx
    End If


    Dim tRow As Long
    tRow = KW_FindFirstBelowInColumnExact(ws, colIdx, totalText, 3)
    If tRow = 0 Then
        ws.Cells(5, colIdx).Value = totalText
        created = True
        LogWrite KW_NowStr() & vbTab & "Поставлена метка 'Итого'" & vbTab & "R5C" & colIdx
    End If
End Sub


Private Function KW_FindFirstInColumnExact(ByVal ws As Worksheet, ByVal col As Long, ByVal whatText As String) As Long
    Dim rng As Range, f As Range
    Set rng = ws.Columns(col)
    Set f = rng.Find(What:=whatText, LookIn:=xlValues, LookAt:=xlWhole, MatchCase:=False)
    If f Is Nothing Then
        KW_FindFirstInColumnExact = 0
    Else
        KW_FindFirstInColumnExact = IIf(f.MergeCells, f.MergeArea.Row, f.Row)
    End If
End Function


Private Function KW_FindFirstBelowInColumnExact(ByVal ws As Worksheet, ByVal col As Long, ByVal whatText As String, ByVal minRow As Long) As Long
    Dim lastRow As Long, rng As Range, f As Range
    lastRow = ws.Rows.Count
    Set rng = ws.Range(ws.Cells(minRow, col), ws.Cells(lastRow, col))
    Set f = rng.Find(What:=whatText, LookIn:=xlValues, LookAt:=xlWhole, MatchCase:=False)
    If f Is Nothing Then
        KW_FindFirstBelowInColumnExact = 0
    Else
        KW_FindFirstBelowInColumnExact = IIf(f.MergeCells, f.MergeArea.Row, f.Row)
    End If
End Function

Private Sub KW_ThickBox(ByVal rng As Range)
    With rng.Borders(xlEdgeLeft): .LineStyle = xlContinuous: .Weight = xlThick: .Color = RGB(0, 0, 0): End With
    With rng.Borders(xlEdgeRight): .LineStyle = xlContinuous: .Weight = xlThick: .Color = RGB(0, 0, 0): End With
    With rng.Borders(xlEdgeTop): .LineStyle = xlContinuous: .Weight = xlThick: .Color = RGB(0, 0, 0): End With
    With rng.Borders(xlEdgeBottom): .LineStyle = xlContinuous: .Weight = xlThick: .Color = RGB(0, 0, 0): End With
End Sub

Private Sub LogWrite(ByVal msg As String)
    On Error Resume Next
    Dim ws As Worksheet, r As Long
    Set ws = ThisWorkbook.Worksheets(LOG_SHEET)
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        ws.Name = LOG_SHEET
        ws.Range("A1:B1").Value = Array("Дата/Время", "Сообщение")
    End If
    r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    ws.Cells(r, 1).Value = Now
    ws.Cells(r, 2).Value = msg
End Sub


Private Function KW_NowStr() As String
    KW_NowStr = Format$(Now, "dd.mm.yyyy HH:nn")
End Function
