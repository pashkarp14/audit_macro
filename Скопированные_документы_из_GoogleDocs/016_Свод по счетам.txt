﻿Option Explicit


'========================================================================================
' Модуль:    modSvod_Accounts
' Версия:    1.1.0  (16.09.2025)
'
' Что нового vs 1.0.0:
'   - Счета (Счет Дт/Счет Кт) выводим с ПОЛНЫМ сохранением вида: колонки для счетов
'     принудительно форматируем как текст ("@"), а значения подготавливаем функцией
'     ToAccountText (без локализационной замены точки на запятую). Это исключает
'     превращение "60.01" в "60,01".
'
' Назначение:
'   Заполняет своды проводок по дебету и кредиту на листе "По счету":
'     - Левый якорь (в кол. 2):  "Свод проводок дебет>>>"
'       Шапка под якорем: "Счет Дт" | "Количество"
'     - Правый якорь (в той же строке правее): "Свод проводок кредит>>>"
'       Шапка под якорем: "Счет Кт" | "Количество"
'   Источник данных: лист(ы) "Популяция для тестирования" и "Популяция для тестирования_Ч*".
'
' Правила:
'   - Перед заполнением очищается область между шапкой и "Итого".
'   - Если строк не хватает — вставляются доп. ячейки ТОЛЬКО в два столбца блока (ShiftDown).
'   - В ячейке "Итого" под столбцом "Количество" — формула СУММ по вставленным строкам.
'   - Формат количества: "# ##0;-# ##0;-".
'   - Сгенерированный блок обводится толстой внешней рамкой.
'========================================================================================


Private Const MacroVersion As String = "1.1.0"


Private Const SHEET_SUMMARY As String = "По счету"
Private Const LOG_SHEET     As String = "Логи"


Private Const POP_BASE      As String = "Популяция для тестирования"
Private Const POP_PART_SUFF As String = "_Ч"


Private Const ANCHOR_DEBIT  As String = "Свод проводок дебет>>>"
Private Const ANCHOR_CREDIT As String = "Свод проводок кредит>>>"


Private Const HEADER_DEBIT_1  As String = "Счет Дт"
Private Const HEADER_DEBIT_2  As String = "Количество"
Private Const HEADER_CREDIT_1 As String = "Счет Кт"
Private Const HEADER_CREDIT_2 As String = "Количество"


Private Const COUNT_NUM_FMT As String = "# ##0;-# ##0;-"
Private Const ANCHOR_COL As Long = 2
Private Const HEADER_HEIGHT As Long = 1
Private Const TOTAL_TEXT As String = "Итого"


Private currentStep As String
Private stepT As Double


'============================== ТОЧКА ВХОДА ====================================


Public Sub Заполнить_Своды_Проводок()
    Dim t0 As Double: t0 = Timer


    Dim appCalc As XlCalculation
    Dim scrUp As Boolean, evt As Boolean, alr As Boolean


    On Error GoTo EH


    appCalc = Application.Calculation
    scrUp = Application.ScreenUpdating
    evt   = Application.EnableEvents
    alr   = Application.DisplayAlerts


    Application.ScreenUpdating = False
    Application.EnableEvents   = False
    Application.DisplayAlerts  = False
    Application.Calculation    = xlCalculationManual


    LogWrite NowStr() & vbTab & "Старт" & vbTab & "Заполнить_Своды_Проводок v" & MacroVersion


    ' ---------- Лист "По счету" и якоря ----------
    StartStep "Поиск листа 'По счету' и якорей"
    Dim wb As Workbook: Set wb = ThisWorkbook
    Dim ws As Worksheet
    On Error Resume Next
    Set ws = wb.Worksheets(SHEET_SUMMARY)
    On Error GoTo 0
    If ws Is Nothing Then Err.Raise 5, , "Лист '" & SHEET_SUMMARY & "' не найден."


    Dim debitAnchorRow As Long, creditAnchorCol As Long
    Dim headerTopDebit As Long, dataTopDebit As Long, totalRowDebit As Long, dataBottomDebit As Long


    debitAnchorRow = FindFirstInColumnExact(ws, ANCHOR_COL, ANCHOR_DEBIT)
    If debitAnchorRow = 0 Then Err.Raise 5, , "Не найден якорь '" & ANCHOR_DEBIT & "' в кол. 2"


    headerTopDebit = debitAnchorRow + 1
    dataTopDebit = headerTopDebit + HEADER_HEIGHT
    totalRowDebit = FindFirstBelowInColumnExact(ws, ANCHOR_COL, TOTAL_TEXT, dataTopDebit)
    If totalRowDebit = 0 Or totalRowDebit <= dataTopDebit Then Err.Raise 5, , "Метка 'Итого' (Дебет) не найдена ниже шапки"
    dataBottomDebit = totalRowDebit - 1


    creditAnchorCol = FindTextInRowExact(ws, debitAnchorRow, ANCHOR_CREDIT, ANCHOR_COL + 1)
    If creditAnchorCol = 0 Then Err.Raise 5, , "Не найден якорь '" & ANCHOR_CREDIT & "' правее дебет-якоря"


    Dim headerTopCredit As Long, dataTopCredit As Long, totalRowCredit As Long, dataBottomCredit As Long
    headerTopCredit = debitAnchorRow + 1
    dataTopCredit = headerTopCredit + HEADER_HEIGHT
    totalRowCredit = FindFirstBelowInColumnExact(ws, creditAnchorCol, TOTAL_TEXT, dataTopCredit)
    If totalRowCredit = 0 Or totalRowCredit <= dataTopCredit Then Err.Raise 5, , "Метка 'Итого' (Кредит) не найдена ниже шапки"
    dataBottomCredit = totalRowCredit - 1


    LogWrite NowStr() & vbTab & "? Якоря" & vbTab & _
        "Дебет: Row=" & debitAnchorRow & ", Col=" & ANCHOR_COL & _
        " | Кредит: Row=" & debitAnchorRow & ", Col=" & creditAnchorCol
    EndStep


    ' ---------- Сбор источников популяции ----------
    StartStep "Сканирование листов популяции"
    Dim popSheets As Collection: Set popSheets = GetPopulationSheets(wb)
    If popSheets Is Nothing Or popSheets.Count = 0 Then
        LogWrite NowStr() & vbTab & "Внимание" & vbTab & "Листы популяции не найдены — своды будут очищены"
    End If
    EndStep


    ' ---------- Агрегация по Дт/Кт ----------
    StartStep "Агрегация (Счет Дт / Счет Кт) из популяции"
    Dim dDt As Object, dKt As Object
    Dim rowsRead As Long
    Set dDt = CreateObject("Scripting.Dictionary"): dDt.CompareMode = vbTextCompare
    Set dKt = CreateObject("Scripting.Dictionary"): dKt.CompareMode = vbTextCompare


    If Not popSheets Is Nothing And popSheets.Count > 0 Then
        AggregateDebitCredit popSheets, dDt, dKt, rowsRead
    End If
    LogWrite NowStr() & vbTab & "? Агрегация" & vbTab & _
        "Прочитано строк=" & rowsRead & _
        "; Уникальных Дт=" & dDt.Count & _
        "; Уникальных Кт=" & dKt.Count
    EndStep


    ' ---------- ДЕБЕТ: очистка/подгонка ----------
    StartStep "ДЕБЕТ: очистка и подгонка строк"
    ClearRangeBorders ws, ws.Range(ws.Cells(dataTopDebit, ANCHOR_COL), ws.Cells(IIf(dataBottomDebit >= dataTopDebit, dataBottomDebit, dataTopDebit), ANCHOR_COL + 1))


    Dim needDebit As Long: needDebit = dDt.Count
    Dim nRowsDebit As Long: nRowsDebit = dataBottomDebit - dataTopDebit + 1
    If needDebit < 0 Then needDebit = 0


    If nRowsDebit < needDebit Then
        Dim insD As Long: insD = needDebit - nRowsDebit
        ws.Range(ws.Cells(totalRowDebit, ANCHOR_COL), ws.Cells(totalRowDebit, ANCHOR_COL + 1)).Resize(insD, 2).Insert xlShiftDown
        totalRowDebit = totalRowDebit + insD
        dataBottomDebit = dataTopDebit + needDebit - 1
        LogWrite NowStr() & vbTab & "ДЕБЕТ: вставлено строк (ячейки в 2 кол.)" & vbTab & insD
    ElseIf nRowsDebit > needDebit Then
        Dim delD As Long: delD = nRowsDebit - needDebit
        If delD > 0 And needDebit >= 0 Then
            ws.Range(ws.Cells(dataTopDebit + needDebit, ANCHOR_COL), ws.Cells(dataTopDebit + needDebit, ANCHOR_COL + 1)).Resize(delD, 2).Delete xlShiftUp
            totalRowDebit = totalRowDebit - delD
            dataBottomDebit = dataTopDebit + needDebit - 1
            LogWrite NowStr() & vbTab & "ДЕБЕТ: удалено лишних строк (ячейки в 2 кол.)" & vbTab & delD
        End If
    Else
        LogWrite NowStr() & vbTab & "ДЕБЕТ: подгонка строк" & vbTab & "Не требуется"
    End If
    EndStep


    ' ---------- КРЕДИТ: очистка/подгонка ----------
    StartStep "КРЕДИТ: очистка и подгонка строк"
    ClearRangeBorders ws, ws.Range(ws.Cells(dataTopCredit, creditAnchorCol), ws.Cells(IIf(dataBottomCredit >= dataTopCredit, dataBottomCredit, dataTopCredit), creditAnchorCol + 1))


    Dim needCredit As Long: needCredit = dKt.Count
    Dim nRowsCredit As Long: nRowsCredit = dataBottomCredit - dataTopCredit + 1
    If needCredit < 0 Then needCredit = 0


    If nRowsCredit < needCredit Then
        Dim insK As Long: insK = needCredit - nRowsCredit
        ws.Range(ws.Cells(totalRowCredit, creditAnchorCol), ws.Cells(totalRowCredit, creditAnchorCol + 1)).Resize(insK, 2).Insert xlShiftDown
        totalRowCredit = totalRowCredit + insK
        dataBottomCredit = dataTopCredit + needCredit - 1
        LogWrite NowStr() & vbTab & "КРЕДИТ: вставлено строк (ячейки в 2 кол.)" & vbTab & insK
    ElseIf nRowsCredit > needCredit Then
        Dim delK As Long: delK = nRowsCredit - needCredit
        If delK > 0 And needCredit >= 0 Then
            ws.Range(ws.Cells(dataTopCredit + needCredit, creditAnchorCol), ws.Cells(dataTopCredit + needCredit, creditAnchorCol + 1)).Resize(delK, 2).Delete xlShiftUp
            totalRowCredit = totalRowCredit - delK
            dataBottomCredit = dataTopCredit + needCredit - 1
            LogWrite NowStr() & vbTab & "КРЕДИТ: удалено лишних строк (ячейки в 2 кол.)" & vbTab & delK
        End If
    Else
        LogWrite NowStr() & vbTab & "КРЕДИТ: подгонка строк" & vbTab & "Не требуется"
    End If
    EndStep


    ' ---------- ДЕБЕТ: вывод данных ----------
    StartStep "ДЕБЕТ: вывод данных и формулы"
    If needDebit > 0 Then
        Dim outD() As Variant: outD = DictToSortedArray2Cols(dDt)
        ' СЧЕТ Дт — текстовый формат, чтобы не было замены точки на запятую
        ws.Range(ws.Cells(dataTopDebit, ANCHOR_COL), ws.Cells(dataBottomDebit, ANCHOR_COL)).NumberFormat = "@"
        ws.Range(ws.Cells(dataTopDebit, ANCHOR_COL), ws.Cells(dataBottomDebit, ANCHOR_COL + 1)).Value = outD
    Else
        If dataBottomDebit >= dataTopDebit Then
            ws.Range(ws.Cells(dataTopDebit, ANCHOR_COL), ws.Cells(dataBottomDebit, ANCHOR_COL + 1)).ClearContents
        End If
    End If


    ' Итого (столбец Количество)
    If dataBottomDebit >= dataTopDebit Then
        Dim rngSumD As Range
        Set rngSumD = ws.Range(ws.Cells(dataTopDebit, ANCHOR_COL + 1), ws.Cells(dataBottomDebit, ANCHOR_COL + 1))
        ws.Cells(totalRowDebit, ANCHOR_COL + 1).FormulaLocal = "=СУММ(" & rngSumD.Address(False, False) & ")"
    Else
        ws.Cells(totalRowDebit, ANCHOR_COL + 1).Value = 0
    End If
    ws.Range(ws.Cells(dataTopDebit, ANCHOR_COL + 1), ws.Cells(totalRowDebit, ANCHOR_COL + 1)).NumberFormatLocal = COUNT_NUM_FMT


    If dataBottomDebit >= dataTopDebit Then
        ThickBox ws.Range(ws.Cells(dataTopDebit, ANCHOR_COL), ws.Cells(dataBottomDebit, ANCHOR_COL + 1))
    End If
    EndStep


    ' ---------- КРЕДИТ: вывод данных ----------
    StartStep "КРЕДИТ: вывод данных и формулы"
    If needCredit > 0 Then
        Dim outK() As Variant: outK = DictToSortedArray2Cols(dKt)
        ' СЧЕТ Кт — текстовый формат
        ws.Range(ws.Cells(dataTopCredit, creditAnchorCol), ws.Cells(dataBottomCredit, creditAnchorCol)).NumberFormat = "@"
        ws.Range(ws.Cells(dataTopCredit, creditAnchorCol), ws.Cells(dataBottomCredit, creditAnchorCol + 1)).Value = outK
    Else
        If dataBottomCredit >= dataTopCredit Then
            ws.Range(ws.Cells(dataTopCredit, creditAnchorCol), ws.Cells(dataBottomCredit, creditAnchorCol + 1)).ClearContents
        End If
    End If


    If dataBottomCredit >= dataTopCredit Then
        Dim rngSumK As Range
        Set rngSumK = ws.Range(ws.Cells(dataTopCredit, creditAnchorCol + 1), ws.Cells(dataBottomCredit, creditAnchorCol + 1))
        ws.Cells(totalRowCredit, creditAnchorCol + 1).FormulaLocal = "=СУММ(" & rngSumK.Address(False, False) & ")"
    Else
        ws.Cells(totalRowCredit, creditAnchorCol + 1).Value = 0
    End If
    ws.Range(ws.Cells(dataTopCredit, creditAnchorCol + 1), ws.Cells(totalRowCredit, creditAnchorCol + 1)).NumberFormatLocal = COUNT_NUM_FMT


    If dataBottomCredit >= dataTopCredit Then
        ThickBox ws.Range(ws.Cells(dataTopCredit, creditAnchorCol), ws.Cells(dataBottomCredit, creditAnchorCol + 1))
    End If
    EndStep


FinishOK:
    LogWrite NowStr() & vbTab & "Готово" & vbTab & _
        "Дебет уник. счетов=" & CountSafe(dDt) & "; Кредит уник. счетов=" & CountSafe(dKt) & _
        "; Время, сек=" & Format$(Timer - t0, "0.00")


Tidy:
    Application.ScreenUpdating = scrUp
    Application.EnableEvents   = evt
    Application.DisplayAlerts  = alr
    Application.Calculation    = appCalc
    Exit Sub


EH:
    LogWrite NowStr() & vbTab & "Ошибка" & vbTab & "Step='" & currentStep & "'; " & Err.Number & " — " & Err.Description
    Resume Tidy
End Sub


'============================== АГРЕГАЦИЯ ПОПУЛЯЦИИ ============================


Private Sub AggregateDebitCredit( _
    ByVal popSheets As Collection, _
    ByRef dDt As Object, _
    ByRef dKt As Object, _
    ByRef rowsRead As Long)


    Dim i As Long, ws As Worksheet
    For i = 1 To popSheets.Count
        Set ws = popSheets(i)
        Dim cDt As Long, cKt As Long, lr As Long
        cDt = FindHeaderCol(ws, "Счет Дт", 1)
        cKt = FindHeaderCol(ws, "Счет Кт", 1)
        If cDt = 0 And cKt = 0 Then GoTo NextSheet


        lr = LastDataRow2(ws)
        If lr < 2 Then GoTo NextSheet


        If cDt > 0 Then
            Dim aDt As Variant
            aDt = ws.Range(ws.Cells(2, cDt), ws.Cells(lr, cDt)).Value2
            rowsRead = rowsRead + UBound(aDt, 1)
            AddCountsFromArray aDt, dDt
        End If


        If cKt > 0 Then
            Dim aKt As Variant
            aKt = ws.Range(ws.Cells(2, cKt), ws.Cells(lr, cKt)).Value2
            If cDt = 0 Then rowsRead = rowsRead + UBound(aKt, 1)
            AddCountsFromArray aKt, dKt
        End If
NextSheet:
    Next i
End Sub


Private Sub AddCountsFromArray(ByVal arr As Variant, ByRef d As Object)
    Dim i As Long, key As String
    For i = 1 To UBound(arr, 1)
        key = ToAccountText(arr(i, 1))
        If LenB(key) > 0 Then
            If d.Exists(key) Then
                d(key) = CLng(d(key)) + 1&
            Else
                d.Add key, 1&
            End If
        End If
    Next i
End Sub


' Нормализуем значение счета в текст без локализационных сюрпризов
Private Function ToAccountText(ByVal v As Variant) As String
    On Error GoTo Fallback
    If IsError(v) Then Exit Function
    If VarType(v) = vbString Then
        ToAccountText = Trim$(CStr(v))
        Exit Function
    End If
    If IsNumeric(v) Then
        Dim s As String
        s = CStr(v)
        ' заменим локальный десятичный разделитель на точку
        s = Replace$(s, Application.International(xlDecimalSeparator), ".")
        ToAccountText = Trim$(s)
        Exit Function
    End If
    ToAccountText = Trim$(CStr(v))
    Exit Function
Fallback:
    ToAccountText = ""
End Function


'============================== ВЫВОД/СОРТИРОВКА ===============================


Private Function DictToSortedArray2Cols(ByVal d As Object) As Variant
    Dim n As Long: n = d.Count
    If n = 0 Then Exit Function
    Dim keys() As String: ReDim keys(1 To n)
    Dim i As Long, k As Variant
    i = 0
    For Each k In d.Keys
        i = i + 1
        keys(i) = CStr(k)
    Next k
    QuickSortStrings keys, 1, n


    Dim outArr() As Variant: ReDim outArr(1 To n, 1 To 2)
    For i = 1 To n
        outArr(i, 1) = keys(i)             ' Счёт (как текст)
        outArr(i, 2) = CLng(d(keys(i)))    ' Кол-во
    Next i
    DictToSortedArray2Cols = outArr
End Function


Private Sub QuickSortStrings(ByRef A() As String, ByVal L As Long, ByVal R As Long)
    Dim i As Long, j As Long
    Dim p As String, t As String
    i = L: j = R: p = A((L + R) \ 2)
    Do While i <= j
        Do While StrComp(A(i), p, vbTextCompare) < 0
            i = i + 1
        Loop
        Do While StrComp(A(j), p, vbTextCompare) > 0
            j = j - 1
        Loop
        If i <= j Then
            t = A(i): A(i) = A(j): A(j) = t
            i = i + 1: j = j - 1
        End If
    Loop
    If L < j Then QuickSortStrings A, L, j
    If i < R Then QuickSortStrings A, i, R
End Sub


'============================== ПОМОЩНИКИ ПО ЛИСТУ =============================


Private Function GetPopulationSheets(ByVal wb As Workbook) As Collection
    Dim col As New Collection, sh As Worksheet, nm As String
    For Each sh In wb.Worksheets
        nm = sh.Name
        If nm = POP_BASE Or Left$(nm, Len(POP_BASE & POP_PART_SUFF)) = POP_BASE & POP_PART_SUFF Then
            col.Add sh
        End If
    Next sh
    Set GetPopulationSheets = col
End Function


Private Function FindTextInRowExact(ByVal ws As Worksheet, ByVal rowNum As Long, ByVal whatText As String, ByVal minCol As Long) As Long
    Dim lastCol As Long, c As Long
    lastCol = ws.Cells(rowNum, ws.Columns.Count).End(xlToLeft).Column
    For c = minCol To lastCol
        If CStr(ws.Cells(rowNum, c).Value) = whatText Then
            FindTextInRowExact = c
            Exit Function
        End If
    Next c
End Function


Private Function FindFirstInColumnExact(ByVal ws As Worksheet, ByVal col As Long, ByVal whatText As String) As Long
    Dim rng As Range, f As Range
    Set rng = ws.Columns(col)
    Set f = rng.Find(What:=whatText, LookIn:=xlValues, LookAt:=xlWhole, MatchCase:=False)
    If f Is Nothing Then
        FindFirstInColumnExact = 0
    Else
        If f.MergeCells Then
            FindFirstInColumnExact = f.MergeArea.Row
        Else
            FindFirstInColumnExact = f.Row
        End If
    End If
End Function


Private Function FindFirstBelowInColumnExact(ByVal ws As Worksheet, ByVal col As Long, ByVal whatText As String, ByVal minRow As Long) As Long
    Dim lastRow As Long, rng As Range, f As Range
    lastRow = ws.Rows.Count
    Set rng = ws.Range(ws.Cells(minRow, col), ws.Cells(lastRow, col))
    Set f = rng.Find(What:=whatText, LookIn:=xlValues, LookAt:=xlWhole, MatchCase:=False)
    If f Is Nothing Then
        FindFirstBelowInColumnExact = 0
    Else
        If f.MergeCells Then
            FindFirstBelowInColumnExact = f.MergeArea.Row
        Else
            FindFirstBelowInColumnExact = f.Row
        End If
    End If
End Function


Private Function FindHeaderCol(ByVal ws As Worksheet, ByVal headerText As String, ByVal hdrRow As Long) As Long
    Dim lastCol As Long, c As Long
    lastCol = ws.Cells(hdrRow, ws.Columns.Count).End(xlToLeft).Column
    For c = 1 To lastCol
        If CStr(ws.Cells(hdrRow, c).Value) = headerText Then
            FindHeaderCol = c
            Exit Function
        End If
    Next c
End Function


Private Function LastDataRow2(ByVal ws As Worksheet) As Long
    Dim lastCol As Long, c As Long, lr As Long, l As Long
    lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column
    For c = 1 To lastCol
        l = ws.Cells(ws.Rows.Count, c).End(xlUp).Row
        If l > lr Then lr = l
    Next c
    If lr < 2 Then lr = 2
    LastDataRow2 = lr
End Function


Private Sub ClearRangeBorders(ByVal ws As Worksheet, ByVal rng As Range)
    On Error Resume Next
    rng.ClearContents
    rng.Borders.LineStyle = xlNone
    On Error GoTo 0
End Sub


Private Sub ThickBox(ByVal rng As Range)
    With rng.Borders(xlEdgeLeft)
        .LineStyle = xlContinuous
        .Weight = xlThick
        .Color = RGB(0, 0, 0)
    End With
    With rng.Borders(xlEdgeRight)
        .LineStyle = xlContinuous
        .Weight = xlThick
        .Color = RGB(0, 0, 0)
    End With
    With rng.Borders(xlEdgeTop)
        .LineStyle = xlContinuous
        .Weight = xlThick
        .Color = RGB(0, 0, 0)
    End With
    With rng.Borders(xlEdgeBottom)
        .LineStyle = xlContinuous
        .Weight = xlThick
        .Color = RGB(0, 0, 0)
    End With
End Sub


Private Function CountSafe(ByVal d As Object) As Long
    On Error Resume Next
    CountSafe = d.Count
End Function


'============================== ЛОГИ ===========================================


Private Sub LogWrite(ByVal msg As String)
    On Error Resume Next
    Dim ws As Worksheet, r As Long
    Set ws = ThisWorkbook.Worksheets(LOG_SHEET)
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        ws.Name = LOG_SHEET
        ws.Range("A1:B1").Value = Array("Дата/Время", "Сообщение")
    End If
    r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    ws.Cells(r, 1).Value = Now
    ws.Cells(r, 2).Value = msg
End Sub


Private Sub StartStep(ByVal stepName As String)
    currentStep = stepName
    stepT = Timer
    LogWrite NowStr() & vbTab & ">> " & stepName
End Sub


Private Sub EndStep()
    LogWrite NowStr() & vbTab & "<< " & currentStep & "  (" & Format$(Timer - stepT, "0.00") & " сек)"
    currentStep = vbNullString
End Sub


Private Function NowStr() As String
    NowStr = Format$(Now, "dd.mm.yyyy HH:nn")
End Function
