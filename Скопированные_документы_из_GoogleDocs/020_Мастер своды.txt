﻿Option Explicit


'========================================================================================
' Модуль:    modMaster_Run
' Версия:    1.0.0  (17.09.2025)
'
' Назначение:
'   Пустой мастер-запуск. Впишите свои вызовы через Call внутри блока
'   "=== ВСТАВЬТЕ СВОИ ВЫЗОВЫ ===".
'
' Правила:
'   - Максимально быстрый режим (ScreenUpdating/Events/Calculation).
'   - Подробные логи на лист "Логи".
'   - Отдельные функции для повторяющихся действий.
'   - Все LogWrite — Private, чтобы не конфликтовать между модулями.
'   - Без однострочных операторов и групповых объявлений переменных.
'========================================================================================


Private Const MacroVersion As String = "1.0.0"
Private Const LOG_SHEET    As String = "Логи"


Private currentStep As String
Private stepT As Double


'============================== ТОЧКА ВХОДА ====================================


Public Sub Мастер_своды()
    Dim t0 As Double
    t0 = Timer


    Dim appCalc As XlCalculation
    appCalc = Application.Calculation


    Dim scrUp As Boolean
    scrUp = Application.ScreenUpdating


    Dim evt As Boolean
    evt = Application.EnableEvents


    Dim alr As Boolean
    alr = Application.DisplayAlerts


    Dim sbPrev As Variant
    sbPrev = Application.StatusBar


    On Error GoTo EH


    SetFastMode True
    Application.StatusBar = "Мастер: запуск v" & MacroVersion & "..."


    LogWrite NowStr() & vbTab & "Старт" & vbTab & "Мастер_Запуск v" & MacroVersion


    ' ======================== ВСТАВЬТЕ СВОИ ВЫЗОВЫ ============================
    ' Пример: каждый логический шаг оборачивайте в StartStep/EndStep и вызывайте через Call
    '
    StartStep "Шаг 1: Свод по счетам"
    Call Заполнить_Свод_по_счетам
    EndStep
    '
    StartStep "Шаг 2: Свод по суммам"
    Call Заполнить_Своды_Сумм
    EndStep
    '
    StartStep "Шаг 3: Свод по дате"
    Call Заполнить_Свод_По_Дате
    EndStep
    '
    StartStep "Шаг 4: Свод по ответственному"
    Call Заполнить_Свод_По_Ответственному
    EndStep
    '
    'StartStep "Шаг 7: Своды — по ответственному"
    'Call modSvod_ByResponsible.Заполнить_Свод_По_Ответственному
    'EndStep
    ' ====================== КОНЕЦ БЛОКА ДЛЯ ВЫЗОВОВ ===========================


FinishOK:
    LogWrite NowStr() & vbTab & "Готово" & vbTab & "Время, сек=" & Format$(Timer - t0, "0.00")


Tidy:
    Application.StatusBar = sbPrev
    RestoreAppSettings scrUp, evt, alr, appCalc
    Exit Sub


EH:
    LogWrite NowStr() & vbTab & "Ошибка" & vbTab & "Step='" & currentStep & "'; " & Err.Number & " — " & Err.Description
    Resume Tidy
End Sub


'============================== СЛУЖЕБНЫЕ ПРОЦЕДУРЫ ============================


Private Sub StartStep(ByVal stepName As String)
    currentStep = stepName
    stepT = Timer
    LogWrite NowStr() & vbTab & ">> " & stepName
End Sub


Private Sub EndStep()
    LogWrite NowStr() & vbTab & "<< " & currentStep & "  (" & Format$(Timer - stepT, "0.00") & " сек)"
    currentStep = vbNullString
End Sub


Private Sub SetFastMode(ByVal enable As Boolean)
    On Error Resume Next
    If enable Then
        Application.ScreenUpdating = False
        Application.EnableEvents = False
        Application.DisplayAlerts = False
        Application.Calculation = xlCalculationManual
    Else
        Application.ScreenUpdating = True
        Application.EnableEvents = True
        Application.DisplayAlerts = True
        Application.Calculation = xlCalculationAutomatic
    End If
    On Error GoTo 0
End Sub


Private Sub RestoreAppSettings( _
    ByVal scrUp As Boolean, _
    ByVal evt As Boolean, _
    ByVal alr As Boolean, _
    ByVal appCalc As XlCalculation)
    On Error Resume Next
    Application.ScreenUpdating = scrUp
    Application.EnableEvents = evt
    Application.DisplayAlerts = alr
    Application.Calculation = appCalc
    On Error GoTo 0
End Sub


'============================== ЛОГИ ===========================================


Private Sub LogWrite(ByVal msg As String)
    On Error Resume Next
    Dim ws As Worksheet
    Dim r As Long


    Set ws = ThisWorkbook.Worksheets(LOG_SHEET)
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        ws.name = LOG_SHEET
        ws.Range("A1:B1").Value = Array("Дата/Время", "Сообщение")
    End If


    r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    ws.Cells(r, 1).Value = Now
    ws.Cells(r, 2).Value = msg
End Sub


Private Function NowStr() As String
    NowStr = Format$(Now, "dd.mm.yyyy HH:nn")
End Function
