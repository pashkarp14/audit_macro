﻿Option Explicit


'========================================================================================
' Модуль:    modSvod_ByResponsible
' Версия:    1.0.0  (17.09.2025)
'
' Назначение:
'   На листе "По ответственному" под якорем "Свод проводок по ответственному>>>"
'   формируется свод из листов популяции:
'     Колонки: Ответственный | Количество | Сумма
'   Источник: "Популяция для тестирования" и "Популяция для тестирования_Ч*".
'   В расчёт берём только строки, где "Ответственный" НЕ пустой.
'
' Особенности:
'   - Быстро: чтение блоками .Value2, агрегация в Scripting.Dictionary, единичная запись.
'   - Автоподгонка числа строк (вставка/удаление), формулы сумм под столбцами,
'     числовой формат "# ##0;-# ##0;-", толстая внешняя рамка результата.
'   - Подробные логи на лист "Логи".
'
' Стиль:
'   - Без однострочных операторов (":") и групповых объявлений через запятую.
'   - Все вспомогательные функции Private (чтобы не конфликтовать с другими модулями).
'========================================================================================


Private Const MacroVersion       As String = "1.0.0"


Private Const SHEET_TARGET       As String = "По ответственному"
Private Const LOG_SHEET          As String = "Логи"


Private Const POP_BASE           As String = "Популяция для тестирования"
Private Const POP_PART_SUFF      As String = "_Ч"


Private Const ANCHOR_COL         As Long = 2
Private Const ANCHOR_TEXT        As String = "Свод проводок по ответственному>>>"
Private Const HEADER_HEIGHT      As Long = 1
Private Const TOTAL_TEXT         As String = "Итого"


Private Const NUM_FMT            As String = "# ##0;-# ##0;-"


Private currentStep As String
Private stepT As Double


'============================== ТОЧКА ВХОДА ====================================


Public Sub Заполнить_Свод_По_Ответственному()
    Dim t0 As Double
    t0 = Timer


    Dim appCalc As XlCalculation
    appCalc = Application.Calculation


    Dim scrUp As Boolean
    scrUp = Application.ScreenUpdating


    Dim evt As Boolean
    evt = Application.EnableEvents


    Dim alr As Boolean
    alr = Application.DisplayAlerts


    On Error GoTo EH


    Application.ScreenUpdating = False
    Application.EnableEvents = False
    Application.DisplayAlerts = False
    Application.Calculation = xlCalculationManual


    LogWrite NowStr() & vbTab & "Старт" & vbTab & "Заполнить_Свод_По_Ответственному v" & MacroVersion


    ' ---------- Лист "По ответственному" и якорь ----------
    StartStep "Поиск листа 'По ответственному' и якоря"
    Dim wb As Workbook
    Set wb = ThisWorkbook


    Dim ws As Worksheet
    On Error Resume Next
    Set ws = wb.Worksheets(SHEET_TARGET)
    On Error GoTo 0
    If ws Is Nothing Then Err.Raise 5, , "Лист '" & SHEET_TARGET & "' не найден."


    Dim anchorRow As Long
    anchorRow = FindFirstInColumnExact(ws, ANCHOR_COL, ANCHOR_TEXT)
    If anchorRow = 0 Then Err.Raise 5, , "Не найден якорь '" & ANCHOR_TEXT & "' в кол. 2"


    Dim headerTop As Long
    headerTop = anchorRow + 1


    Dim dataTop As Long
    dataTop = headerTop + HEADER_HEIGHT


    Dim totalRow As Long
    totalRow = FindFirstBelowInColumnExact(ws, ANCHOR_COL, TOTAL_TEXT, dataTop)
    If totalRow = 0 Or totalRow <= dataTop Then Err.Raise 5, , "Метка 'Итого' не найдена ниже шапки"


    Dim dataBottom As Long
    dataBottom = totalRow - 1


    LogWrite NowStr() & vbTab & "? Якорь" & vbTab & _
        "AnchorRow=" & anchorRow & "; HeaderTop=" & headerTop & "; DataTop=" & dataTop & "; TotalRow=" & totalRow
    EndStep


    ' ---------- Источники (ПОПУЛЯЦИЯ) ----------
    StartStep "Поиск листов популяции"
    Dim popSheets As Collection
    Set popSheets = GetPopulationSheets(wb)
    If popSheets Is Nothing Or popSheets.Count = 0 Then
        LogWrite NowStr() & vbTab & "Внимание" & vbTab & "Листы популяции не найдены — свод будет очищен"
    Else
        LogWrite NowStr() & vbTab & "Найдено листов популяции" & vbTab & popSheets.Count
    End If
    EndStep


    ' ---------- Агрегация по ответственным ----------
    StartStep "Агрегация (Ответственный -> Количество, Сумма)"
    Dim dVals As Object
    Set dVals = CreateObject("Scripting.Dictionary")
    dVals.CompareMode = vbTextCompare


    Dim dDisp As Object
    Set dDisp = CreateObject("Scripting.Dictionary")
    dDisp.CompareMode = vbTextCompare


    Dim rowsRead As Long
    Dim rowsWithResp As Long
    Dim rowsWithSumParsed As Long


    If Not popSheets Is Nothing And popSheets.Count > 0 Then
        AggregateByResponsible_FromPopulation popSheets, dVals, dDisp, rowsRead, rowsWithResp, rowsWithSumParsed
    End If


    LogWrite NowStr() & vbTab & "? Агрегация" & vbTab & _
        "Прочитано строк=" & rowsRead & _
        "; С ответственным=" & rowsWithResp & _
        "; С распознанной суммой=" & rowsWithSumParsed & _
        "; Уникальных ответственных=" & dVals.Count
    EndStep


    ' ---------- Очистка / Подгонка строк (3 столбца) ----------
    StartStep "Очистка и подгонка количества строк (3 столбца)"
    ClearRangeBorders ws, ws.Range(ws.Cells(dataTop, ANCHOR_COL), ws.Cells(IIf(dataBottom >= dataTop, dataBottom, dataTop), ANCHOR_COL + 2))


    Dim need As Long
    need = dVals.Count


    Dim nRows As Long
    nRows = dataBottom - dataTop + 1
    If need < 0 Then need = 0


    If nRows < need Then
        Dim insN As Long
        insN = need - nRows
        ws.Range(ws.Cells(totalRow, ANCHOR_COL), ws.Cells(totalRow, ANCHOR_COL + 2)).Resize(insN, 3).Insert xlShiftDown
        totalRow = totalRow + insN
        dataBottom = dataTop + need - 1
        LogWrite NowStr() & vbTab & "Вставлено строк (ячейки в 3 кол.)" & vbTab & insN
    ElseIf nRows > need Then
        Dim delN As Long
        delN = nRows - need
        If delN > 0 Then
            ws.Range(ws.Cells(dataTop + need, ANCHOR_COL), ws.Cells(dataTop + need, ANCHOR_COL + 2)).Resize(delN, 3).Delete xlShiftUp
            totalRow = totalRow - delN
            dataBottom = dataTop + need - 1
            LogWrite NowStr() & vbTab & "Удалено лишних строк (ячейки в 3 кол.)" & vbTab & delN
        End If
    Else
        LogWrite NowStr() & vbTab & "Подгонка строк" & vbTab & "Не требуется"
    End If
    EndStep


    ' ---------- Вывод данных + формулы ----------
    StartStep "Вывод данных и формул"
    If need > 0 Then
        Dim outArr As Variant
        outArr = DictToSortedArrayResp(dVals, dDisp)


        ' Ответственный — текст; Кол-во/Сумма — формат чисел
        ws.Range(ws.Cells(dataTop, ANCHOR_COL), ws.Cells(dataBottom, ANCHOR_COL)).NumberFormat = "@"
        ws.Range(ws.Cells(dataTop, ANCHOR_COL + 1), ws.Cells(totalRow, ANCHOR_COL + 2)).NumberFormatLocal = NUM_FMT


        ws.Range(ws.Cells(dataTop, ANCHOR_COL), ws.Cells(dataBottom, ANCHOR_COL + 2)).Value = outArr
    Else
        If dataBottom >= dataTop Then
            ws.Range(ws.Cells(dataTop, ANCHOR_COL), ws.Cells(dataBottom, ANCHOR_COL + 2)).ClearContents
        End If
    End If


    ' Формулы СУММ под столбцами Количество и Сумма
    If dataBottom >= dataTop Then
        Dim rngCnt As Range
        Dim rngSum As Range
        Set rngCnt = ws.Range(ws.Cells(dataTop, ANCHOR_COL + 1), ws.Cells(dataBottom, ANCHOR_COL + 1))
        Set rngSum = ws.Range(ws.Cells(dataTop, ANCHOR_COL + 2), ws.Cells(dataBottom, ANCHOR_COL + 2))
        ws.Cells(totalRow, ANCHOR_COL + 1).FormulaLocal = "=СУММ(" & rngCnt.Address(False, False) & ")"
        ws.Cells(totalRow, ANCHOR_COL + 2).FormulaLocal = "=СУММ(" & rngSum.Address(False, False) & ")"
    Else
        ws.Cells(totalRow, ANCHOR_COL + 1).Value = 0
        ws.Cells(totalRow, ANCHOR_COL + 2).Value = 0
        ws.Range(ws.Cells(totalRow, ANCHOR_COL + 1), ws.Cells(totalRow, ANCHOR_COL + 2)).NumberFormatLocal = NUM_FMT
    End If


    ' Рамка
    If dataBottom >= dataTop Then
        ThickBox ws.Range(ws.Cells(dataTop, ANCHOR_COL), ws.Cells(dataBottom, ANCHOR_COL + 2))
    End If
    EndStep


FinishOK:
    LogWrite NowStr() & vbTab & "Готово" & vbTab & _
        "Уникальных ответственных=" & CountSafe(dVals) & _
        "; Время, сек=" & Format$(Timer - t0, "0.00")


Tidy:
    Application.ScreenUpdating = scrUp
    Application.EnableEvents = evt
    Application.DisplayAlerts = alr
    Application.Calculation = appCalc
    Exit Sub


EH:
    LogWrite NowStr() & vbTab & "Ошибка" & vbTab & "Step='" & currentStep & "'; " & Err.Number & " — " & Err.Description
    Resume Tidy
End Sub


'============================== АГРЕГАЦИЯ (из популяции) =======================


Private Sub AggregateByResponsible_FromPopulation( _
    ByVal sheets As Collection, _
    ByRef dVals As Object, _          ' key=canon, value=Array(count, sum)
    ByRef dDisp As Object, _          ' key=canon, value=first display form
    ByRef rowsRead As Long, _
    ByRef rowsWithResp As Long, _
    ByRef rowsWithSumParsed As Long)


    Dim i As Long
    For i = 1 To sheets.Count
        Dim ws As Worksheet
        Set ws = sheets(i)


        Dim colResp As Long
        Dim colSum As Long
        Dim lr As Long


        Dim hdrResp As String
        Dim hdrSum As String


        colResp = FindHeaderSyn(ws, Array("Ответственный", "ответственный", "ответств", "ответственный "), 1, hdrResp)
        colSum = FindHeaderSyn(ws, Array("Сумма", " сумма", "сумма ", "Сумма абс", "Сумма абс.", "Сумма абсолютная", "Сумма (абс)"), 1, hdrSum)


        If colResp = 0 Or colSum = 0 Then
            LogWrite NowStr() & vbTab & "Пропуск листа популяции (нет Ответственный/Сумма)" & vbTab & ws.Name & _
                    "  Ответственный=" & colResp & "  Сумма=" & colSum
            GoTo NextSheet
        End If


        lr = LastDataRowOnCols(ws, Array(colResp, colSum))
        If lr < 2 Then GoTo NextSheet


        Dim minC As Long
        Dim maxC As Long
        If colResp < colSum Then
            minC = colResp
            maxC = colSum
        Else
            minC = colSum
            maxC = colResp
        End If


        Dim block As Variant
        block = ws.Range(ws.Cells(1, minC), ws.Cells(lr, maxC)).Value2


        Dim respIdx As Long
        Dim sumIdx As Long
        respIdx = colResp - minC + 1
        sumIdx = colSum - minC + 1


        Dim r As Long
        Dim respRaw As String
        Dim canon As String
        Dim disp As String


        Dim v As Double
        Dim ok As Boolean
        Dim pair As Variant


        rowsRead = rowsRead + (lr - 1)
        For r = 2 To lr
            respRaw = VToStr(block(r, respIdx))
            respRaw = Trim$(Replace$(respRaw, Chr$(160), " "))
            If LenB(respRaw) = 0 Then GoTo NextR


            canon = CanonTextKey(respRaw)
            disp = respRaw
            rowsWithResp = rowsWithResp + 1


            ok = False
            v = ToDoubleSafe2(block(r, sumIdx), ok)
            If ok Then
                rowsWithSumParsed = rowsWithSumParsed + 1
            Else
                v = 0#
            End If


            If dVals.Exists(canon) Then
                pair = dVals(canon)
                pair(0) = CLng(pair(0)) + 1&
                pair(1) = CDbl(pair(1)) + v
                dVals(canon) = pair
            Else
                pair = Array(1&, CDbl(v))     ' {count, sum}
                dVals.Add canon, pair
                If Not dDisp.Exists(canon) Then dDisp.Add canon, disp
            End If
NextR:
        Next r


        LogWrite NowStr() & vbTab & "Источник популяции" & vbTab & ws.Name & _
                "  Строк=" & (lr - 1) & _
                "  ОтветственныйКолонка='" & hdrResp & "'(c" & colResp & ")" & _
                "  СуммаКолонка='" & hdrSum & "'(c" & colSum & ")"
NextSheet:
    Next i
End Sub


'============================== ВЫВОД / СОРТИРОВКА =============================


' dVals: key=canon -> Array(count, sum)
' dDisp: key=canon -> display text
Private Function DictToSortedArrayResp(ByVal dVals As Object, ByVal dDisp As Object) As Variant
    Dim n As Long
    n = dVals.Count
    If n = 0 Then Exit Function


    Dim keys() As String
    ReDim keys(1 To n)


    Dim i As Long
    Dim k As Variant
    i = 0
    For Each k In dVals.Keys
        i = i + 1
        keys(i) = CStr(k)
    Next k


    QuickSortStrings keys, 1, n


    Dim outArr() As Variant
    ReDim outArr(1 To n, 1 To 3)


    Dim pair As Variant
    For i = 1 To n
        pair = dVals(keys(i))
        outArr(i, 1) = dDisp(keys(i))        ' Ответственный (первая встреченная форма)
        outArr(i, 2) = CLng(pair(0))         ' Количество
        outArr(i, 3) = CDbl(pair(1))         ' Сумма
    Next i


    DictToSortedArrayResp = outArr
End Function


Private Sub QuickSortStrings(ByRef A() As String, ByVal L As Long, ByVal R As Long)
    Dim i As Long
    Dim j As Long
    Dim p As String
    Dim t As String


    i = L
    j = R
    p = A((L + R) \ 2)


    Do While i <= j
        Do While A(i) < p
            i = i + 1
        Loop
        Do While A(j) > p
            j = j - 1
        Loop
        If i <= j Then
            t = A(i)
            A(i) = A(j)
            A(j) = t
            i = i + 1
            j = j - 1
        End If
    Loop


    If L < j Then QuickSortStrings A, L, j
    If i < R Then QuickSortStrings A, i, R
End Sub


'============================== ПОМОЩНИКИ ПО ЛИСТАМ ============================


Private Function GetPopulationSheets(ByVal wb As Workbook) As Collection
    Dim col As New Collection
    Dim sh As Worksheet
    Dim nm As String


    For Each sh In wb.Worksheets
        nm = sh.Name
        If nm = POP_BASE Or Left$(nm, Len(POP_BASE & POP_PART_SUFF)) = POP_BASE & POP_PART_SUFF Then
            col.Add sh
        End If
    Next sh


    Set GetPopulationSheets = col
End Function


Private Function FindFirstInColumnExact(ByVal ws As Worksheet, ByVal col As Long, ByVal whatText As String) As Long
    Dim rng As Range
    Dim f As Range


    Set rng = ws.Columns(col)
    Set f = rng.Find(What:=whatText, LookIn:=xlValues, LookAt:=xlWhole, MatchCase:=False)


    If f Is Nothing Then
        FindFirstInColumnExact = 0
    Else
        If f.MergeCells Then
            FindFirstInColumnExact = f.MergeArea.Row
        Else
            FindFirstInColumnExact = f.Row
        End If
    End If
End Function


Private Function FindFirstBelowInColumnExact(ByVal ws As Worksheet, ByVal col As Long, ByVal whatText As String, ByVal minRow As Long) As Long
    Dim lastRow As Long
    Dim rng As Range
    Dim f As Range


    lastRow = ws.Rows.Count
    Set rng = ws.Range(ws.Cells(minRow, col), ws.Cells(lastRow, col))
    Set f = rng.Find(What:=whatText, LookIn:=xlValues, LookAt:=xlWhole, MatchCase:=False)


    If f Is Nothing Then
        FindFirstBelowInColumnExact = 0
    Else
        If f.MergeCells Then
            FindFirstBelowInColumnExact = f.MergeArea.Row
        Else
            FindFirstBelowInColumnExact = f.Row
        End If
    End If
End Function


' Поиск заголовка по набору синонимов. Возвращает найденную колонку и исходный текст заголовка.
Private Function FindHeaderSyn(ByVal ws As Worksheet, ByVal syns As Variant, ByVal hdrRow As Long, ByRef foundHeader As String) As Long
    Dim lastCol As Long
    lastCol = ws.Cells(hdrRow, ws.Columns.Count).End(xlToLeft).Column


    Dim j As Long
    Dim i As Long
    Dim head As String
    Dim key As String


    For j = 1 To lastCol
        head = NormalizeHeaderText(CStr(ws.Cells(hdrRow, j).Value))
        If head <> "" Then
            For i = LBound(syns) To UBound(syns)
                key = NormalizeHeaderText(CStr(syns(i)))
                If head = key Then
                    foundHeader = CStr(ws.Cells(hdrRow, j).Value)
                    FindHeaderSyn = j
                    Exit Function
                End If
            Next i
        End If
    Next j
End Function


Private Function LastDataRowOnCols(ByVal ws As Worksheet, ByVal cols As Variant) As Long
    Dim lr As Long
    Dim j As Long
    Dim lrj As Long
    Dim c As Long


    lr = 0
    For j = LBound(cols) To UBound(cols)
        c = CLng(cols(j))
        If c > 0 Then
            lrj = ws.Cells(ws.Rows.Count, c).End(xlUp).Row
            If lrj > lr Then lr = lrj
        End If
    Next j


    If lr < 2 Then lr = 2
    LastDataRowOnCols = lr
End Function


Private Function NormalizeHeaderText(ByVal s As String) As String
    Dim nb As String
    nb = Chr$(160)


    s = Replace$(s, nb, " ")
    s = Replace$(s, vbCr, " ")
    s = Replace$(s, vbLf, " ")
    s = Replace$(s, vbTab, " ")
    s = Replace$(s, "ё", "е")
    s = Replace$(s, "Ё", "Е")
    s = LCase$(Trim$(s))


    Do While InStr(s, "  ") > 0
        s = Replace$(s, "  ", " ")
    Loop


    NormalizeHeaderText = s
End Function


Private Sub ClearRangeBorders(ByVal ws As Worksheet, ByVal rng As Range)
    On Error Resume Next
    rng.ClearContents
    rng.Borders.LineStyle = xlNone
    On Error GoTo 0
End Sub


Private Sub ThickBox(ByVal rng As Range)
    With rng.Borders(xlEdgeLeft)
        .LineStyle = xlContinuous
        .Weight = xlThick
        .Color = RGB(0, 0, 0)
    End With
    With rng.Borders(xlEdgeRight)
        .LineStyle = xlContinuous
        .Weight = xlThick
        .Color = RGB(0, 0, 0)
    End With
    With rng.Borders(xlEdgeTop)
        .LineStyle = xlContinuous
        .Weight = xlThick
        .Color = RGB(0, 0, 0)
    End With
    With rng.Borders(xlEdgeBottom)
        .LineStyle = xlContinuous
        .Weight = xlThick
        .Color = RGB(0, 0, 0)
    End With
End Sub


Private Function CountSafe(ByVal d As Object) As Long
    On Error Resume Next
    CountSafe = d.Count
End Function


'============================== ТЕКСТ/ЧИСЛО ====================================


Private Function VToStr(ByVal v As Variant) As String
    On Error Resume Next
    VToStr = CStr(v)
    On Error GoTo 0
End Function


' Ключ для группировки по тексту (канонизация пробелов/регистра)
Private Function CanonTextKey(ByVal s As String) As String
    Dim nb As String
    nb = Chr$(160)
    s = Replace$(s, nb, " ")
    s = Trim$(s)
    Do While InStr(s, "  ") > 0
        s = Replace$(s, "  ", " ")
    Loop
    CanonTextKey = LCase$(s)
End Function


' Надёжное чтение суммы (пробелы/неразрывные/запятая)
Private Function ToDoubleSafe2(ByVal v As Variant, ByRef ok As Boolean) As Double
    On Error GoTo Fallback


    ok = False
    If IsError(v) Then Exit Function


    If IsNumeric(v) Then
        ok = True
        ToDoubleSafe2 = CDbl(v)
        Exit Function
    End If


    Dim s As String
    Dim nb As String


    s = CStr(v)
    If LenB(s) = 0 Then Exit Function


    nb = Chr$(160)
    s = Replace$(s, nb, "")
    s = Replace$(s, " ", "")
    s = Replace$(s, ",", Application.International(xlDecimalSeparator))


    ToDoubleSafe2 = CDbl(s)
    ok = True
    Exit Function
Fallback:
    ok = False
    ToDoubleSafe2 = 0#
End Function


'============================== ЛОГИ ===========================================


Private Sub LogWrite(ByVal msg As String)
    On Error Resume Next
    Dim ws As Worksheet
    Dim r As Long


    Set ws = ThisWorkbook.Worksheets(LOG_SHEET)
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        ws.Name = LOG_SHEET
        ws.Range("A1:B1").Value = Array("Дата/Время", "Сообщение")
    End If


    r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    ws.Cells(r, 1).Value = Now
    ws.Cells(r, 2).Value = msg
End Sub


Private Sub StartStep(ByVal stepName As String)
    currentStep = stepName
    stepT = Timer
    LogWrite NowStr() & vbTab & ">> " & stepName
End Sub


Private Sub EndStep()
    LogWrite NowStr() & vbTab & "<< " & currentStep & "  (" & Format$(Timer - stepT, "0.00") & " сек)"
    currentStep = vbNullString
End Sub


Private Function NowStr() As String
    NowStr = Format$(Now, "dd.mm.yyyy HH:nn")
End Function
