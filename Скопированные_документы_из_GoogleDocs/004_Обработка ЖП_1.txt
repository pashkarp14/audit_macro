﻿Option Explicit


'========================================================================================
' Модуль:    modJP_BulkProcess_NoFilters
' Версия:    1.3.1  (15.09.2025)
'
' Что нового:
'   1.3.1 — Переименован заголовок выходной колонки "Меппинг" -> "Связанные?".
'
' Что нового в 1.3.0 (ускорение + «как в оригинале» по форматам):
'   - Вернулся к массивам: 1 чтение источника, 1 запись результата (мгновенно на больших объёмах).
'   - Счета Дт/Кт — текстовые столбцы ("@") с установкой формата ДО записи (точки не превращаются в запятые).
'   - Для остальных 11 колонок переношу формат столбца (NumberFormat) от источника (проценты/даты сохраняются).
'   - «Проводка» и «Связанные?» формируются из массива (без обращения к ячейкам).
'   - Результирующий лист для каждого ЖП_Ч ставится справа от исходного.
'========================================================================================


Private Const MODULE_VERSION As String = "1.3.1"


' Имена/префиксы
Private Const SRC_NAME_PART As String = "ЖП_Ч"
Private Const OUT_PREFIX     As String = "ЖП_обработанный_"
Private Const LOG_SHEET      As String = "Логи"


' ---------------- Нужные 13 заголовков (порядок вывода) ----------------------
' Итог: 13 базовых + "Проводка" + "Связанные?" = 15 колонок
Private Function RequiredHeaders() As Variant
    RequiredHeaders = Array( _
        "N", _
        "Дата", _
        "Документ", _
        "Счет Дт", _
        "Субконто1 Дт", _
        "Субконто2 Дт", _
        "Субконто3 Дт", _
        "Счет Кт", _
        "Субконто1 Кт", _
        "Субконто2 Кт", _
        "Субконто3 Кт", _
        "Сумма", _
        "Содержание" _
    )
End Function


'============================== ТОЧКА ВХОДА ====================================
Public Sub Обработать_Все_ЖП_Ч_БезФильтров()
    Dim t0 As Double: t0 = Timer
    Private_StartPerf
    On Error GoTo EH
    
    Private_Log "Старт v" & MODULE_VERSION
    
    Dim wb As Workbook: Set wb = ThisWorkbook
    
    ' 1) Источники
    Dim srcSheets As Collection
    Set srcSheets = GetJPSheets(wb, SRC_NAME_PART)
    If srcSheets Is Nothing Or srcSheets.Count = 0 Then
        Private_Log "Источник: листы с '" & SRC_NAME_PART & "' не найдены"
        GoTo Tidy
    End If
    Private_Log "Найдено листов-источников: " & srcSheets.Count
    
    ' 2) Подготовка VGO-словаря (может быть пустым)
    Dim wsVGO As Worksheet: Set wsVGO = GetSheetByName("Список ВГО")
    Dim vgoCol As Long: vgoCol = FindVGOCol(wsVGO)
    Dim dictVGO As Object: Set dictVGO = BuildDictionary(wsVGO, vgoCol)
    Private_Log "VGO: кол=" & vgoCol & "; уникальных=" & dictVGO.Count
    
    ' 3) Основной цикл
    Dim req As Variant: req = RequiredHeaders()
    Dim reqCnt As Long: reqCnt = UBound(req) - LBound(req) + 1
    
    Dim i As Long
    For i = 1 To srcSheets.Count
        Dim wsSrc As Worksheet: Set wsSrc = srcSheets(i)
        Dim wsOut As Worksheet: Set wsOut = GetOrCreateProcessedAfter(wsSrc, OUT_PREFIX & i)
        ProcessOneSheet wsSrc, wsOut, req, reqCnt, dictVGO
    Next i
    
    ' 4) Почистить лишние обработанные
    DeleteExtraProcessedSheets wb, OUT_PREFIX, srcSheets.Count
    
Tidy:
    Private_FinishPerf t0, "Готово"
    Exit Sub


EH:
    Private_Log "ОШИБКА [" & Err.Number & "]: " & Err.Description
    Private_FinishPerf t0, "С ошибкой"
End Sub


'============================== ОБРАБОТКА ОДНОГО ЛИСТА ========================
Private Sub ProcessOneSheet(ByVal wsSrc As Worksheet, ByVal wsOut As Worksheet, ByVal req As Variant, ByVal reqCnt As Long, ByVal dictVGO As Object)
    Private_Log ">> " & wsSrc.Name & " -> " & wsOut.Name
    
    ' --- Карта заголовков по строке 1 ---
    Dim headerMap As Object: Set headerMap = BuildHeaderMap(wsSrc)
    
    ' --- Найти колонки источника для нужных полей ---
    Dim srcCols() As Long
    ReDim srcCols(1 To reqCnt)
    
    Dim k As Long, miss As String, c As Long
    For k = 1 To reqCnt
        c = FindHeaderBySynonyms(headerMap, CStr(req(k - 1 + LBound(req))))
        srcCols(k) = c
        If c = 0 Then
            If Len(miss) > 0 Then miss = miss & ", "
            miss = miss & """" & CStr(req(k - 1 + LBound(req))) & """"
        End If
    Next k
    If Len(miss) > 0 Then
        Private_Log "Пропуск '" & wsSrc.Name & "': нет заголовков: " & miss
        Exit Sub
    End If
    
    ' --- Последняя строка данных (максимум по найденным колонкам) ---
    Dim lr As Long
    lr = LastDataRow(wsSrc, srcCols)
    If lr < 2 Then
        Private_Log "Нет данных на '" & wsSrc.Name & "' (только шапка)"
        wsOut.Cells.Clear
        Exit Sub
    End If
    
    ' --- Чтение одним блоком (min..max) ---
    Dim minCol As Long, maxCol As Long
    minCol = Application.WorksheetFunction.Min(srcCols)
    maxCol = Application.WorksheetFunction.Max(srcCols)
    
    Dim srcBlock As Variant
    srcBlock = wsSrc.Range(wsSrc.Cells(1, minCol), wsSrc.Cells(lr, maxCol)).Value2
    
    ' --- Относительные индексы в блоке ---
    Dim relIdx() As Long
    ReDim relIdx(1 To reqCnt)
    Dim iRel As Long
    For iRel = 1 To reqCnt
        relIdx(iRel) = srcCols(iRel) - minCol + 1
    Next iRel
    
    ' --- Формируем выходной массив: 15 колонок ---
    Dim outCols As Long: outCols = reqCnt + 2 ' + Проводка + Связанные?
    Dim outArr() As Variant
    ReDim outArr(1 To (lr - 1) + 1, 1 To outCols) ' с шапкой
    
    ' Шапка
    Dim outCol As Long
    outCol = 0
    For k = 1 To reqCnt
        outCol = outCol + 1
        outArr(1, outCol) = req(k - 1 + LBound(req))
    Next k
    outCol = outCol + 1: outArr(1, outCol) = "Проводка"
    outCol = outCol + 1: outArr(1, outCol) = "Связанные?"
    
    ' Позиции в ВЫХОДЕ (фикс. порядок RequiredHeaders):
    Const POS_ACC_DT As Long = 4
    Const POS_ACC_KT As Long = 8
    Const POS_SUB1_KT As Long = 9
    Const POS_SUM As Long    = 12 ' на случай форматирования суммы
    
    ' Наполнение из источника + вычисление Проводка/Связанные?
    Dim r As Long, rr As Long
    Dim deb As String, cred As String, sub1kt As String
    rr = 2
    For r = 2 To lr
        ' 13 базовых полей
        outCol = 0
        For k = 1 To reqCnt
            outCol = outCol + 1
            outArr(rr, outCol) = srcBlock(r, relIdx(k))
        Next k
        ' Проводка
        deb = CStr(outArr(rr, POS_ACC_DT))
        cred = CStr(outArr(rr, POS_ACC_KT))
        outArr(rr, reqCnt + 1) = "Дт " & deb & " Кт " & cred
        ' Связанные? (по Субконто1 Кт через словарь ВГО)
        sub1kt = Trim$(CStr(outArr(rr, POS_SUB1_KT)))
        If Len(sub1kt) > 0 And dictVGO.Exists(sub1kt) Then
            outArr(rr, reqCnt + 2) = "ВГО"
        Else
            outArr(rr, reqCnt + 2) = "третьи стороны"
        End If
        rr = rr + 1
    Next r
    
    ' --- Выгрузка одним присваиванием + минимум форматирования (как в «оригинале») ---
    With wsOut
        .Cells.Clear
        .AutoFilterMode = False
        
        ' 1) Сразу ставим ТЕКСТОВЫЙ формат для счетов (до записи массивов)
        .Columns(POS_ACC_DT).NumberFormat = "@"
        .Columns(POS_ACC_KT).NumberFormat = "@"
        
        ' 2) Переносим формат столбца от источника по остальным требуемым колонкам
        '    (быстро: одно свойство на колонку; проценты/даты сохраняются как в источнике)
        Dim kCol As Long
        For kCol = 1 To reqCnt
            If kCol <> POS_ACC_DT And kCol <> POS_ACC_KT Then
                On Error Resume Next
                .Columns(kCol).NumberFormat = wsSrc.Columns(srcCols(kCol)).NumberFormat
                On Error GoTo 0
            End If
        Next kCol
        
        ' 3) Заливка массива значений
        .Range("A1").Resize(UBound(outArr, 1), outCols).Value = outArr
        
        ' 4) Шапка/внешний вид
        .Range("A1").Resize(1, outCols).AutoFilter
        .Rows(1).Font.Bold = True
        .UsedRange.WrapText = False
        .Columns("A:O").EntireColumn.AutoFit ' 15 столбцов
    End With
    
    Private_Log "<< Готово: " & wsSrc.Name & " — строк=" & (lr - 1) & "; столбцов=15"
End Sub


'============================== ЛИСТЫ: ПОИСК/ПОРЯДОК ===========================
Private Function GetJPSheets(ByVal wb As Workbook, ByVal namePart As String) As Collection
    Dim col As New Collection, sh As Worksheet
    For Each sh In wb.Worksheets
        If InStr(1, sh.Name, namePart, vbTextCompare) > 0 Then col.Add sh
    Next sh
    Set GetJPSheets = col
End Function


' Создать/получить лист результата и переместить его СПРАВА от источника
Private Function GetOrCreateProcessedAfter(ByVal wsSrc As Worksheet, ByVal outName As String) As Worksheet
    Dim wb As Workbook: Set wb = wsSrc.Parent
    Dim ws As Worksheet: Set ws = GetSheetByName(outName)
    If ws Is Nothing Then
        Set ws = wb.Worksheets.Add(After:=wsSrc)
        ws.Name = outName
        Private_Log "Создан справа: " & outName
    Else
        On Error Resume Next
        ws.Move After:=wsSrc
        On Error GoTo 0
        Private_Log "Перемещён справа: " & outName
    End If
    Set GetOrCreateProcessedAfter = ws
End Function


' Удалить лишние ЖП_обработанный_k (k > needCount)
Private Sub DeleteExtraProcessedSheets(ByVal wb As Workbook, ByVal prefix As String, ByVal needCount As Long)
    Dim sh As Worksheet, idx As Long, nm As String
    For Each sh In wb.Worksheets
        nm = sh.Name
        If Left$(nm, Len(prefix)) = prefix Then
            idx = Val(Mid$(nm, Len(prefix) + 1))
            If idx > needCount Then
                Private_Log "Удаляю лишний: " & nm
                Application.DisplayAlerts = False
                sh.Delete
                Application.DisplayAlerts = True
            End If
        End If
    Next sh
End Sub


Private Function GetSheetByName(ByVal nm As String) As Worksheet
    On Error Resume Next
    Set GetSheetByName = ThisWorkbook.Worksheets(nm)
    On Error GoTo 0
End Function


'============================== ЗАГОЛОВКИ: НОРМАЛИЗАЦИЯ ========================
Private Function BuildHeaderMap(ByVal ws As Worksheet) As Object
    Dim d As Object: Set d = CreateObject("Scripting.Dictionary")
    d.CompareMode = vbTextCompare
    Dim lastCol As Long, c As Long, raw As String, norm As String
    lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column
    For c = 1 To lastCol
        raw = CStr(ws.Cells(1, c).Value)
        norm = NormalizeHeaderText(raw)
        If Len(norm) > 0 Then If Not d.Exists(norm) Then d.Add norm, c
    Next c
    Set BuildHeaderMap = d
End Function


Private Function FindHeaderBySynonyms(ByVal map As Object, ByVal logicalName As String) As Long
    Dim syns As Variant: syns = HeaderSynonyms(logicalName)
    Dim i As Long, key As String
    For i = LBound(syns) To UBound(syns)
        key = NormalizeHeaderText(CStr(syns(i)))
        If map.Exists(key) Then
            FindHeaderBySynonyms = map(key)
            Exit Function
        End If
    Next i
End Function


Private Function HeaderSynonyms(ByVal logicalName As String) As Variant
    Select Case LCase$(logicalName)
        Case "n":               HeaderSynonyms = Array("n", "№", "номер")
        Case "дата":            HeaderSynonyms = Array("дата")
        Case "документ":        HeaderSynonyms = Array("документ")
        Case "счет дт":         HeaderSynonyms = Array("счет дт", "счёт дт")
        Case "субконто1 дт":    HeaderSynonyms = Array("субконто1 дт", "субконто 1 дт")
        Case "субконто2 дт":    HeaderSynonyms = Array("субконто2 дт", "субконто 2 дт")
        Case "субконто3 дт":    HeaderSynonyms = Array("субконто3 дт", "субконто 3 дт")
        Case "счет кт":         HeaderSynonyms = Array("счет кт", "счёт кт")
        Case "субконто1 кт":    HeaderSynonyms = Array("субконто1 кт", "субконто 1 кт")
        Case "субконто2 кт":    HeaderSynonyms = Array("субконто2 кт", "субконто 2 кт")
        Case "субконто3 кт":    HeaderSynonyms = Array("субконто3 кт", "субконто 3 кт")
        Case "сумма":           HeaderSynonyms = Array("сумма", " сумма", "сумма ")
        Case "содержание":      HeaderSynonyms = Array("содержание")
        Case Else:              HeaderSynonyms = Array(logicalName)
    End Select
End Function


Private Function NormalizeHeaderText(ByVal s As String) As String
    Dim nb As String: nb = Chr$(160)
    s = Replace$(s, nb, " ")
    s = Replace$(s, vbCr, " ")
    s = Replace$(s, vbLf, " ")
    s = Replace$(s, vbTab, " ")
    s = Replace$(s, "ё", "е")
    s = Replace$(s, "Ё", "Е")
    s = Trim$(s)
    Do While InStr(s, "  ") > 0
        s = Replace$(s, "  ", " ")
    Loop
    NormalizeHeaderText = LCase$(s)
End Function


'============================== ПОСЛЕДНЯЯ СТРОКА ===============================
' cols — Variant (массив номеров колонок)
Private Function LastDataRow(ByVal ws As Worksheet, ByVal cols As Variant) As Long
    On Error GoTo SafeExit
    Dim lr As Long, j As Long, lrj As Long, colNum As Long
    If IsArray(cols) Then
        For j = LBound(cols) To UBound(cols)
            colNum = CLng(cols(j))
            If colNum > 0 Then
                lrj = ws.Cells(ws.Rows.Count, colNum).End(xlUp).Row
                If lrj > lr Then lr = lrj
            End If
        Next j
    Else
        colNum = CLng(cols)
        If colNum > 0 Then lr = ws.Cells(ws.Rows.Count, colNum).End(xlUp).Row
    End If
    If lr = 0 Then lr = ws.UsedRange.Rows.Count
    If lr < 2 Then lr = 2
    LastDataRow = lr
    Exit Function
SafeExit:
    LastDataRow = 2
End Function


'============================== VGO / Дикты ====================================
Private Function BuildDictionary(ws As Worksheet, colNum As Long) As Object
    Dim d As Object: Set d = CreateObject("Scripting.Dictionary")
    d.CompareMode = vbTextCompare
    If ws Is Nothing Or colNum <= 0 Then Set BuildDictionary = d: Exit Function
    Dim lastRow As Long: lastRow = ws.Cells(ws.Rows.Count, colNum).End(xlUp).Row
    If lastRow < 2 Then Set BuildDictionary = d: Exit Function
    Dim v As Variant, i As Long, s As String
    v = ws.Range(ws.Cells(2, colNum), ws.Cells(lastRow, colNum)).Value2
    If IsArray(v) Then
        For i = 1 To UBound(v, 1)
            s = Trim$(CStr(v(i, 1))): If LenB(s) > 0 Then d(s) = True
        Next i
    Else
        s = Trim$(CStr(v)): If LenB(s) > 0 Then d(s) = True
    End If
    Set BuildDictionary = d
End Function


Private Function FindVGOCol(ws As Worksheet) As Long
    Dim lastCol As Long, j As Long, txt As String
    If ws Is Nothing Then Exit Function
    lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column
    For j = 1 To lastCol
        txt = LCase$(CStr(ws.Cells(1, j).Value))
        If InStr(txt, "контрагент") > 0 Then FindVGOCol = j: Exit Function
    Next j
End Function


'============================== ЛОГИ/ПЕРФ ======================================
Private Sub Private_Log(ByVal msg As String)
    Dim ws As Worksheet, r As Long
    On Error Resume Next
    Set ws = ThisWorkbook.Worksheets(LOG_SHEET)
    On Error GoTo 0
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        ws.Name = LOG_SHEET
        ws.Range("A1:C1").Value = Array("Дата/Время", "Источник/Версия", "Сообщение")
    End If
    r = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    ws.Cells(r, 1).Value = Now
    ws.Cells(r, 2).Value = "BulkJP v" & MODULE_VERSION
    ws.Cells(r, 3).Value = msg
End Sub


Private Sub Private_StartPerf()
    On Error Resume Next
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    Application.DisplayAlerts = False
    Application.Calculation = xlCalculationManual
    On Error GoTo 0
End Sub


Private Sub Private_FinishPerf(ByVal t0 As Double, ByVal finalMsg As String)
    On Error Resume Next
    Application.ScreenUpdating = True
    Application.EnableEvents = True
    Application.DisplayAlerts = True
    Application.Calculation = xlCalculationAutomatic
    On Error GoTo 0
    Private_Log finalMsg & " — " & Format$(Timer - t0, "0.00") & " c"
End Sub
